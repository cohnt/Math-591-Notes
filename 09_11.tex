\defn{
	Let $M$ be a topological manifold. A \u{$C^{\infty}$ atlas} on $M$, $\mathscr{A}=\set{(U_{i},\phi_{i})}_{i\in{}I}$, is a collection of charts on $M$ such that $M=\bigcup_{i\in{}I}U_{i}$ and $\forall{}i,j\in{}I$, $(U_{i},\phi_{i})$ and $(U_{j},\phi_{j})$ are $C^{\infty}$ compatible.\n
}

\par\noindent
Last time, we considered $M=S^{1}\subseteq\R^{2}$ with the subspace topology. We constructed two atlases, $\mathscr{A}$ and $\mathscr{B}$, and claimed every chart in $\mathscr{B}$ is compatible with every chart in $\mathscr{A}$, i.e., $\mathscr{A}\cup\mathscr{B}$ is a $C^{\infty}$ atlas.\n

\par\noindent
We want a definition of smooth structure on $M$ that includes $\mathscr{A}$ and $\mathscr{B}$.\n

\defn{
	A \u{smooth structure} on a topological manifold $M$ is a \u{maximal smooth atlas}, $\mathscr{M}$, i.e., a smooth atlas such that:\n
	If $(U,\phi)$ is any continuous chart on $M$ \st{} $\forall(V,\psi)\in\mathscr{M}$, $(U,\phi)$ and $(V,\psi)$ are compatible, then $(U,\phi)\in\mathscr{M}$.\n
}

\prop{
	If $\mathscr{A}$ is a smooth atlas on $M$, $\exists\unique\mathscr{M}$, a maximal smooth atlas, \st{} $\mathscr{A}\subseteq\mathscr{M}$.\n
	Proof: Let $\mathscr{M}=\set{(U,\phi)\ptxt{ continuous chart \st{} }\forall(V,\psi)\in\mathscr{A},(U,\phi)\ptxt{ and }(V,\psi)\ptxt{ are compatible}}$. We need to show $\mathscr{M}$ is a smooth atlas, and that it is maximal.\n
	First, check that it's a smooth atlas. Clearly $\mathscr{A}\subseteq\mathscr{M}$. So charts in $\mathscr{M}$ cover $M$. We just need to show compatibility. Let $(U_{1},\phi_{1})$ and $(U_{2},\phi_{2})$ be charts in $\mathscr{M}$. We need to show they're compatible with each other. Let\break $p\in{}U_{1}\cap{}U_{2}$, and show $\phi_{2}\of\phi_{1}\inv$ is $C^{\infty}$ at $\phi_{1}(p)$. Well, $\exists{}(V,\psi)\in\mathscr{A}$ \st{} $p\in{}V$. And every $(U_{i},\phi_{i})$ is compatible with charts in $\mathscr{A}$, so $\phi_{2}\of\phi_{1}\inv=(\phi_{2}\of\psi\inv)\of(\psi\of\phi_{1}\inv)$ on some neighborhood of $\phi_{1}(p)$. This is a smooth map, so we conclude $\mathscr{M}$ is a smooth atlas.\n
	Proving it's maximal just uses topology, and is left as an exercise.\n
	\proven
}

\defn{
	A \u{differentiable manifold} is a pair $(M,\mathscr{M})$, where $M$ is a topological manifold, and $\mathscr{M}$ is a maximal $C^{\infty}$ atlas.\n
}

\ex{
	$M=\R$, $\mathscr{M}$ is the maximal atlas containing $(\R,x\mapsto\sqrt[3]{x})$.\n
	\i{This is a different smooth structure on $\R$.} However, the standard $\R$ and this $\R$ are isomorphic in the category of smooth manifolds: they're diffeomorphic! The isomorphism is $x\mapsto\sqrt[3]{x}$.\n
}

\par\noindent
Question (very hard): Given a topological manifold $M$, are there non-isomorphic smooth structures on $M$? Yes, it can happen!\n

\par\noindent
Consider $U\subseteq\R^{N}$ open (recall: $\R^{N}$ is called the ambient space), and $F:U\to\R^{k}$ smooth.\n
$F=(f^{1},\ldots,f^{k})$, with $f^{i}:U\to\R$ smooth.\n
Recall: If $x_{0}\in{}U$, the Jacobian of $F$ at $x_{0}$ is the matrix
\[
	J(F)(x_{0})=F'(x_{0})=\begin{pmatrix}
		\horizontalMatrixLine & \nabla{}f^{1}(x_{0}) & \horizontalMatrixLine\\
		& \vdots &\\
		\horizontalMatrixLine & \nabla{}f^{k}(x_{0}) & \horizontalMatrixLine
	\end{pmatrix}
\]

\defn{
	$0\in\R^{k}$ is a \u{regular value} of $F$ iff $\forall{}x\in{}F\inv(0)$, $F'(x)$ has rank $k$. (I.e. it defines a surjective linear map onto $\R^{k}$.)\n
}

\[
	F'=\left(\vphantom{\begin{matrix}
		\pd{f^{1}}{x_{1}} & \cdots & \pd{f^{1}}{x_{N}}\\
		\vdots &  & \vdots\\
		\pd{f^{k}}{x_{1}} & \cdots & \pd{f^{k}}{x_{N}}
	\end{matrix}}\right.\underbrace{\begin{matrix}
		\pd{f^{1}}{x_{1}} & \cdots & \pd{f^{1}}{x_{N}}\\
		\vdots &  & \vdots\\
		\pd{f^{k}}{x_{1}} & \cdots & \pd{f^{k}}{x_{N}}
	\end{matrix}}_{N}\left.\vphantom{\begin{matrix}
		\pd{f^{1}}{x_{1}} & \cdots & \pd{f^{1}}{x_{N}}\\
		\vdots &  & \vdots\\
		\pd{f^{k}}{x_{1}} & \cdots & \pd{f^{k}}{x_{N}}
	\end{matrix}}\right)\left.\vphantom{\begin{matrix}
		\pd{f^{1}}{x_{1}} & \cdots & \pd{f^{1}}{x_{N}}\\
		\vdots &  & \vdots\\
		\pd{f^{k}}{x_{1}} & \cdots & \pd{f^{k}}{x_{N}}
	\end{matrix}}\right\}k
\]

\newpage
\thm{
	Assume $0\in\R^{k}$ is a regular value of $F$. Then $\forall{}x_{0}\in{}F\inv(0)$, there's some neighborhood $W$ of $x_{0}$ in the ambient space \st{} $F\inv(0)\cap{}W$ is the graph of a function of $N-k$ variables into $\R^{k}$, up to a permutation of the coordinates $x_{1},\ldots,x_{N}$. $^{*}$This is just the implicit function theorem!$^{*}$\n
}

\ex{
	An example of ``permutation of coordinates''.\n
	Let $k=2$ and $N=5$ (so $N-k=3$). Let $A\subseteq\R^{3}$ open, $G=(g^{1},g^{2}):A\to\R^{2}$. Then
	\[
		F\inv(0)\cap{}W=\set{(x_{1},x_{2},x_{3},x_{4},x_{5})\mid{}x_{2}=g^{1}(x_{1},x_{3},x_{4}),x_{5}=g^{2}(x_{1},x_{3},x_{4})}
	\]
}
