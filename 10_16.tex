\subsection*{Smooth Sections of Vector Bundles}

\par\noindent
Start with a rank $\rho$ vector bundle, with section $s$, i.e., $\pi\of{}s=I_{M}$.
\[
	\begin{tikzcd}
		\mathcal{E} \arrow[d, "\pi"]\\
		M \arrow[u, bend right=45, "s", swap]
	\end{tikzcd}
\]
Let $\chi$ be a local trivialization
\[
	\begin{tikzcd}
		\pi\inv(U) \arrow[rr, "\chi"] \arrow[dr] &[-10pt] &[-10pt] U\times\R^{\rho} \arrow[dl]\\
		& U
	\end{tikzcd}
\]
with $\chi$ a diffeomorphism, and linear on each fiber. Then $\restr{s}{U}:U\to\pi\inv(U)$ satisfies
\[
	\map{\chi\of(\restr{s}{U}):U}{U\times\R^{\rho}}{p}{(p,F(p))}
\]
where $F:U\to\R^{\rho}$. We write $F=(F^{1},\ldots,F^{\rho})$ with each $F^{i}:U\to\R$.\n

\lemma{
	(From last time) $\restr{s}{U}$ is smooth iff $\forall{}i$, $F^{i}$ is smooth.\nn
	Proof: $\Rightarrow$ is trivial.\n
	\phantom{Proof: }$\Leftarrow$: It's a fact from analysis that $F$ is $C^{\infty}$ iff $\forall{}i$, $F^{i}$ is $C^{\infty}$. So $\restr{s}{U}(p)=\chi\inv(p,F(p))$, which is smooth.\n
	\proven
}

\par\noindent
Observe: The trivializationabove corresponds to a ``moving frame'' on $U$.\n

\defn{
	A \u{moving frame} on $U$ i a collection of $\rho$ smooth sections on $U$, $\set{e_{1},\ldots,e_{\rho}}$, \st{} $\forall{}p\in{}U$, $\set{e_{1}(p),\ldots,e_{\rho}(p)}$ is a basis of the fiber $\pi\inv(p)$.\n
}

\ex{
	If $(U,\phi=(x^{1},\ldots,x^{n}))$ is a coordinate chart, let $e_{i}(p)=\pdat{}{x^{i}}{p}\in{}T_{p}M$. This defines a moving frame of $TM$ on $U$.\n
}

\par\noindent
Given a trivialization $\chi$ over $U$ as above, how do we get a moving frame? Well, $\forall{}i\in\set{1,\ldots,\rho},p\in{}U$, let\n
$e_{i}(p)\eqdef\chi\inv(p,(0,\ldots,1,\ldots0))$ (with the $1$ in the $i$th entry).\n

\par\noindent
Observe: If $\restr{s}{U}$ corresponds to $F=(F^{1},\ldots,F^{\rho}):U\to\R^{\rho}$, then $\restr{s}{U}=\sum_{i=1}^{\rho}F^{i}e_{i}$, where the $F^{i}$ are scalar-valued functions and the $e_{i}$ are sections. So $\forall{}p\in{}U$, $s(p)\in\pi\inv(p)$, and $s(p)=\sum_{i=1}^{\rho}F^{i}(p)e_{i}(p)$ (using the vector space structure of $\pi\inv(p)$).\n

\par\noindent
Conversely, we can also define a trivialization from a moving frame. (This is left as an exercise.)\n

\par\noindent
Observe: If $C^{\infty}(M,\mathcal{E})$ is the space of $C^{\infty}$ sections of $\mathcal{E}\to{}M$ vector bundles, then $C^{\infty}(M,\mathcal{E})$ is a module over $C^{\infty}(M)$. We can multiply a section $s$ by a function $f\in{}C^{\infty}(M)$ fiber-wise, with $(fs)(p)=f(p)s(p)$.\n

\subsection*{Vector Fields}

\par\noindent
Let $\mathcal{E}=TM$.\n

\defn{
	$\mathfrak{X}$ is the set of all smooth vector fields on $M$.\n
}

\par\noindent
$\forall{}X\in\mathfrak{X}$, with a coordinate system on $U$, $\exists{}a_{i}\in{}C^{\infty}(U)$ \st{} $X=\sum_{i=1}^{n}a_{i}\pd{}{x^{i}}$. $X$ is $C^{\infty}$ iff $\forall{}i,a_{i}\in{}C^{\infty}$.\n

\prop{
	Any $X\in\mathfrak{X}(M)$ defines an operator
	\[
		\map{C^{\infty}(M)}{C^{\infty}(M)}{f}{X(f)}
	\]
	which
	\begin{enumerate}[label=\alph*), topsep=0pt, itemsep=0pt, leftmargin=4\parindent]
		\item is $\R$-linear.
		\item satisfies Leibniz' rule: $\forall{}f,g\in{}C^{\infty}(M)$, $X(fg)=fX(g)+gX(f)$.
	\end{enumerate}\up\nn
	(An aside: As a section, the value of $X$ at $p\in{}M$ is denoted $X_{p}\in{}T_{p}M$.)\nn
	Proof: $X(f)(p)=X_{p}([f])$, where $[f]$ is the germ of $f$ at $p$. Thus, $X$ is a derivation on $C^{\infty}(M)$, because $X_{p}$ is a derivation on $C_{p}^{\infty}(M)$ germs.\proven
}

\defn{
	Such an operator is called a \u{derivation} of $C^{\infty}(M)$.\n
}

\prop{
	(1) The operator defined by $X\in\mathfrak{X}(M)$ is local, i.e., $\forall{}f\in{}C^{\infty}(M),U\subseteq{}M$ open such that $\restr{f}{U}\equiv{}0$, then $\restr{X(f)}{U}\equiv{}0$.\n
}

\par\noindent
Observe: This ``locality'' characterizes differential operators.\n

\par\noindent
Observe: In local coordinates, if $X=\sum_{i}a_{i}\pd{}{x^{i}}$, then $X(f)(p)=\sum_{i}a_{i}\pd{f}{x^{i}}(p)$.\n

\thm{
	(2) Any operator $D:C^{\infty}(M)\to{}C^{\infty}(M)$ that is a derivation is given by a vector field.\n
}

\thm{
	(3) The commutator of two derivations is a derivation.\n
}

\par\noindent
Together, we have: If $X,Y\in\mathfrak{X}(M)$, then there is a vector field denoted $[X,Y]\in\mathfrak{X}(M)$ (said ``$X$ bracket $Y$'' or ``$X$ commutator $Y$'') such that $\forall{}f\in{}C^{\infty}(M)$, $[X,Y](f)=X(Y(f))-Y(X(f))$.\n

\par\noindent
Proof of (3): Define $[X,Y]$ as the operator commutator above. Clearly this is linear. Verify Leibniz' rule:
\[
	[X,Y](fg)=X(fY(g)+gY(f))-Y(fX(g)+gX(f))=\cdots=f[X,Y](g)+g[X,Y](f)
\]
\proven

\par\noindent
In local coordinates, say $X=\sum_{i}a_{i}\pd{}{x^{i}}$ and $Y=\sum_{j}b_{j}\pd{}{x^{j}}$. Then
\[
	[X,Y]=\sum_{ij}\brack{a_{i}\pd{}{x^{i}},b_{j}\pd{}{x^{j}}}
\]
And
\begin{align*}
	\brack{a_{i}\pd{}{x^{i}},b_{j}\pd{}{x^{j}}} & =a_{i}\pd{}{x^{i}}(b_{j}\pd{f}{x^{j}})-b_{j}\pd{}{x^{j}}(a_{i}\pd{f}{x^{i}})\\
	& =\cancel{a_{i}b_{j}\frac{\partial^{2}f}{\partial{}x^{i}\partial{}x^{j}}}+a_{i}\pd{b_{j}}{x^{j}}\pd{f}{x^{j}}-\paren{\cancel{b_{j}a_{i}\frac{\partial^{2}f}{\partial{}x^{j}\partial{}x^{i}}}+b_{j}\pd{a_{i}}{x^{j}}\pd{f}{x^{i}}}\\
	& =a_{i}\pd{b_{j}}{x^{i}}\pd{f}{x^{j}}-b_{j}\pd{a_{i}}{x^{j}}\pd{f}{x^{i}}
\end{align*}
This gives the commutator.
