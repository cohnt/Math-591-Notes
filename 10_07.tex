\subsection*{The Hopf Fibration}

\[
	\begin{tikzcd}
		\C^{2} &[-15pt] S^{3} \arrow[l, symbol=\subset] \arrow[d, "\pi"]\\
		& \CP^{1} \arrow[r, symbol=\cong] &[-15pt] S^{1}\backslash{}S^{3}
	\end{tikzcd}
\]

\par\noindent
$\pi(z_{0},z_{1})=\set{(e^{i\theta}z_{0},e^{i\theta}z_{1})\mid{}\theta\in\R/2\pi\Z}$.\n
$\pi(z_{0},z_{1})=[z_{0}:z_{1}]$.\n

\par\noindent
Claim: This is a fiber bundle with fiber $\Phi=S^{1}$.\n

\par\noindent
For the covering of $\CP^{1}$, we choose the same covering as used in the homework: $U_{0}=\set{[z_{0}:z_{1}]\mid{}z_{0}\ne{}0}$ and\break $U_{1}=\set{[z_{0}:z_{1}]:z_{1}\ne{}0}$.\n

\par\noindent
Define $\mathscr{S}(\mathfrak{z})=\frac{1}{\sqrt{1+\abs{\mathfrak{z}}}}(1,\mathfrak{z})$. Then we have
\[
	\begin{tikzcd}
		&[-25pt] & & S^{3} \arrow[d,"\pi"]\\
		\R^{2} \arrow[r, symbol=\cong] & \C & U_{0} \arrow[l, "\cong", swap] \arrow[r, hook] \arrow[ur, "\mathscr{S}"] & \CP^{1}\\[-15pt]
		& \mathfrak{z}=\frac{z_{1}}{z_{0}} & {[z_{0}:z_{1}]} \arrow[l, mapsto]
	\end{tikzcd}
\]
Note: $\pi\of\mathscr{S}=I_{U_{0}}$, since $\pi(\mathscr{S}(\mathfrak{z}))=\mathfrak{z}$.\n

\defn{
	If $\pi:M\to{}N$ is a fiber bundle, a \u{section} of $\pi$ is a $C^{\infty}$ map $\mathscr{S}:N\to{}M$ \st{} $\pi\of\mathscr{S}=I_{N}$.\n
}

\par\noindent
Observe: The Hopf fibration does not have a global section.\n

\par\noindent
Observe: Local sections always exist, because they always exist for the trivial bundle $N\to{}N\times\Phi$: fix $\nu\in\Phi$, and map $p\mapsto(p,\nu)$.\n

\par\noindent
More on the Hopf fibration... Define a trivialization
\[
	\begin{tikzcd}
		\pi\inv(U_{0}) \arrow[rr,"\chi"] \arrow[dr] &[-15pt] &[-15pt] U_{0}\times\Phi \arrow[dl]\\
		& U_{0}
	\end{tikzcd}
\]
with $\chi(\mathfrak{z},e^{i\theta})=e^{i\theta}\mathscr{S}(\mathfrak{z})=\frac{e^{i\theta}}{\sqrt{1+\abs{\mathfrak{z}}}}(1,\mathfrak{z})$. Then $\pi\of\mathscr{S}=I_{U_{0}}$, and $\forall\mathfrak{z}\in{}U_{0}$, $\mathscr{S}(\mathfrak{z})=\pi\inv(\mathfrak{z})$.\n

\par\noindent
$\chi(z_{0},z_{1})=(\frac{z_{1}}{z_{0}},\frac{z_{0}}{\abs{z_{0}}})$ (note that $z_{0}\ne{}0$). Then $S^{3}\cong\R^{3}\cup\set{\infty}=\bigsqcup{}S^{1}$, an uncountable disjoint union.\n

\subsection*{Vector Bundles}

\defn{
	$\pi:E\to{}B$ is a \u{vector bundle} with \u{base} $B$ and \u{rank} $k\in\N$ iff
	\begin{enumerate}[itemsep=0pt, topsep=0pt, label=\alph*)]
		\item It's a fibration.
		\item $\forall{}p\in{}B$, the fiber $\pi\inv(p)$ is a vector space.
		\item There is an open covering $\set{U_{\alpha}}$ of $B$ and all trivializations $\chi_{\alpha}:\pi\inv(U_{\alpha})\to{}U_{\alpha}\times\R^{k}$ restrict to linear maps on each fiber, i.e.,
		\[
			\begin{tikzcd}[ampersand replacement=\&]
				\restr{\chi_{\alpha}}{\pi\inv(p)}:\pi\inv(p) \arrow[r,"\cong"] \arrow[rr, bend right=30, "\ptxt{Linear Isomorphism}", swap] \& \set{p}\times\R^{k} \arrow[r,symbol=\cong] \&[-15pt] \R^{k}
			\end{tikzcd}
		\]
	\end{enumerate}\up\n
}

\par\noindent
Note: $\forall{}p\in{}B$, the zero section applied to $p\in{}B$ gives $0\in\pi\inv(p)$.\n

\ex{
	\begin{enumerate}[label=\arabic*), leftmargin=3\parindent, topsep=0pt]
		\item The tangent bundle $TB\to{}B$.
		\item The cotangent bundle $\displaystyle{}T^{*}B=\bigsqcup_{p\in{}B}\set{p}\times{}T_{p}^{*}B\to{}B$.
	\end{enumerate}\up\n
}

\defn{
	If $S\subseteq{}M$ is a submanifold, then the \u{co-normal bundle of $S$} is $\mathcal{N}=\set{(p,\alpha)\in{}T^{*}M\mid{}\restr{\alpha}{T_{p}S}=0}$.\n
}

\par\noindent
The co-normal bundle is a vector bundle.\n

\par\noindent
Note: If we give $T_{p}M$ a Euclidean inner product, we can identify
\[
	\begin{tikzcd}
		T_{p}^{*}M \arrow[r,symbol=\cong] &[-15pt] T_{p}M\\[-20pt]
		\iprod{\,\cdot\,,v} & v \arrow[l, mapsto]
	\end{tikzcd}
\]

\par\noindent
Claim: $\mathcal{N}\subset{}T^{*}M$ is a submanifold, and $\pi:\mathcal{N}\to{}S$ is a vector bundle of rank $\codim{}S=\dim{}M-\dim{}S$. The fiber of $\mathcal{N}$ over $p$ is the annihilator of $T_{p}S$.\n
