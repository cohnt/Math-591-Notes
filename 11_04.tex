\section*{Lie Groups and Their Algebras}

\par\noindent
Reminder/Review: Given $G$ a Lie group, $\forall{}g\in{}G$, the map $L_{g}:G\to{}G$ where $L_{g}(k)=gk$. $X\in\mathfrak{X}(M)$ is left-invariant iff $\forall{}g\in{}G$, $X$ is $L_{g}$-related to itself.\n

\prop{
	(HW 8 Problem 4) There is a bijective linear correspondence between $\mathfrak{g}=T_{e}G$, the Lie algebra, and the set of left-invariant fields on $G$, where $T_{e}G\ni{}A\mapsto{}A^{\sharp}\in\mathfrak{X}(G)$. $A^{\sharp}$ is defined by $\forall{}g\in{}G$, $A^{\sharp}_{g}=(L_{g})_{*,e}(A)$. $A^{\sharp}$ is smooth.\n
}

\par\noindent
Observe: $\forall{}X,Y\in\mathfrak{X}(G)$ left-invariant, $\brack{X,Y}$ is also left-invariant, because being related by $L_{g}$ preserves commutators.\n

\defn{
	Under this correspondence, we can define the \u{bracket of fields}
	\[
		\map{\brack{\,\cdot\,,\,\cdot\,}:\mathfrak{g}\times\mathfrak{g}}{\mathfrak{g}}{(A,B)}{\brack{A,B}\eqdef\brack{A^{\sharp},B^{\sharp}}_{e}}
	\]
}

\defn{
	$(g,\brack{\,\cdot\,,\,\cdot\,}$ is the \u{Lie algebra} of $G$.\n
}

\par\noindent
$\brack{\,\cdot\,,\,\cdot\,}$ is $\R$-bilinear and satisfies the Jacobi identity.\n

\section*{The Exponential Map}

\par\noindent
Notation: $\forall{}A\in{}\mathfrak{g}$, let $F^{A}$ be the flow of $A^{\sharp}$.\n

\defn{
	$\forall{}A\in\mathfrak{g}$, the \u{exponential map} is defined to be $\exp{}t\,A\eqdef{}F^{A}_{t}(e)$.\n
}

\prop{
	Given $A\in\mathfrak{g}$:
	\begin{enumerate}[label=(\arabic*), leftmargin=4\parindent]
		\item $\exp{}t\,A$ is defined $\forall{}t\in\R$.
		\item $\exp(t+s)\,A=(\exp{}t\,A)\cdot(\exp{}s\,A)$, $\forall{}s,t\in\R$ (with $\cdot$ being group multiplication).
	\end{enumerate}\n
	Proof (2): Assume $t+s$ is small. Then
	\begin{align*}
		& \exp{}(t+s)\,A=F^{A}_{t+s}(e)=F^{A}_{t}(F^{A}_{s}(e))\\
		& (\exp{}t\,A)\cdot(\exp{}s\,A)=L_{\exp{}t\,A}(\exp{}s\,A)
	\end{align*}
	So $L_{\exp{}t\,A}$ maps integral curves of $A^{\sharp}$ to integral curves of $A^{\sharp}$, because $A^{\sharp}$ is $L_{\exp{}t\,A}$-related to itself. Thus, the map $s\mapsto{}L_{\exp{}t\,A}(\exp{}s\,A)$ is the integral cuve of $A^{\sharp}$ through $\exp{}t\,A$, so it must agree with $F^{A}_{s}(\exp{}t\,A)$. This proves (2) for small $s,t$.\proven\up\nn
	Proof (1): Well, we know $\exists\varepsilon>0$ \st{} $\exp{}t\,A$ is defined for $t\in(-\varepsilon,\varepsilon)$. So we'll make use of the fact that\break $\exp(t+s)\,A=(\exp{}t\,A)\cdot(\exp{}s\,A)$. Note: the right-hand side is defined for $t+s\in(-\varepsilon,\varepsilon)$, so extend the left-hand side to $t+s\in(-2\varepsilon,2\varepsilon)$. This is somewhat sketchy, but it works. Then, we just have to check that this extension is an integral curve of $A^{\sharp}$, and it must agree with $\exp(t+s)\,A$. Now, we have $\exp{}t\,A$ defined for $t\in(-2\varepsilon,2\varepsilon)$. Repeat ad nauseum...\proven
}

\cor{
	(2) $\Rightarrow$ $\exp{}t\,A,\exp{}s\,A\in{}G$ commute.\n
}

\ex{
	$G=\GL(n,\R)\osubseteq\R^{n^{2}}$. $\mathfrak{g}=\operatorname{gl}(n,\R)=\R^{n^{2}}$, the set of $n\times{}n$ real matrices. Then
	\[
		\exp{}\,A=\sum_{n=0}^{\infty}\frac{1}{n!}t^{n}A^{n}
	\]
	We need to check that this series converges absolutely (i.e. for some matrix norm). Well, $\norm{AB}\le\norm{A}\norm{B}$, and $\frac{d}{dt}(\exp{}t\,A)=A\exp{}t\,A=(\exp{}t\,A)A=A^{\sharp}A$.\n
	(Claim: $\forall{}g\in\GL(n,\R)$, $L_{g}(A)=A^{\sharp}g$. Proof: $L_{g}:\R^{n^{2}}\to\R^{n^{2}}$ is linear, so its differential is itself, i.e., $(L_{g})_{*,e}=L_{g}$.)\n
}

\defn{
	$\exp:\mathfrak{g}\to{}G$ is defined by $\exp(A)\eqdef\restr{\exp(t)\,A}{t=1}$.\n
}

\prop{
	$(\exp)_{*,0}:T_{0}\mathfrak{g}\to\mathfrak{g}$ is the identity map $\mathfrak{g}\to\mathfrak{g}$, so $\exp$ is a local diffeomorphism at $0\in\mathfrak{g}$.\nn
	Proof:
	\[
		(\exp)_{*,0}(A)\labeledeq{(1)}\restr{\frac{d}{dt}\exp{}t\,A}{t=0}=A^{\sharp}_{e}=A
	\]
	where (1) holds by using the curve $t\mapsto{}tA$, in $\mathfrak{g}$ adapted to $(0,A)$.\proven
}

\prop{
	$\forall{}A\in\mathfrak{g}$, $A^{\sharp}$ is complete.\nn
	Proof: $\forall{}g\in{}G$, $L_{g}(\exp{}t\,A)=g\cdot\exp{}t\,A$ is the integral curve of $A^{\sharp}$ starting at $g$.\proven
}

\subsection*{Subgroups (Part 1)}

\defn{
	A \u{regular} (or \u{closed}, or \u{embedded}) \u{subgroup} $H$ of $G$ is a regular submanifold that is also a subgroup. It follows directly that $H$ is a lie group in its own right, and $\mathfrak{h}=T_{e}H\hookrightarrow{}\mathfrak{g}=T_{e}G$.\n
}

\prop{
	$\mathfrak{h}$ is closed under $\brack{\,\cdot\,,\,\cdot\,}$ of $\mathfrak{g}$. This means, $\forall{}A,B\in\mathfrak{h}$, $\brack{A^{\sharp},B^{\sharp}}$ is tangent to $H$, and $\brack{A^{\sharp},B^{\sharp}}_{e}\in\mathfrak{h}$.\n
}
