\defn{
	A space $X$ is \underline{locally Euclidean} iff every point in $X$ has a neighborhood homeomorphic to $\R^{n}$, for some fixed $n$.\n
}

\defn{
	A \underline{topological manifold} is a space that is locally Euclidean, Hausdorff, and second countable.\n
}

\thm{
	If $U\subseteq\R^{n}$ and $V\subseteq\R^{m}$ are homeomorphic nonempty open sets, then $m=n$. In other words, ``dimension is topological''.\n
	\n
	The idea of this proof is to show that any open set in $\R$ can be covered by families of open sets with overlaps of at most $2$ sets, any open set in $\R^{2}$ can be covered by families of open sets with overlaps of at most $3$ sets, and so on.\n
}

\par\noindent
Observe that in the definition of locally Euclidean, it's equivalent to ask that $\forall{}p\in{}X$, $p$ has a neighborhood homeomorphic to an open subset of $\R^{n}$.\n

\defn{
	Let $M$ be a topological manifold. If $U\subseteq{}M$ is open, and $\phi:U\to\R^{n}$ is a homeomorphism onto an open set $\phi(U)\subseteq\R^{n}$, then the pair $(U,\phi)$ is a \underline{chart} of $M$.\n
}

\defn{
	Let $(U,\phi)$ and $(V,\psi)$ be charts, with $U\cap{}V\ne\emptyset$. The \underline{transition function} (from $\phi$ to $\psi$) is a map
	\[
		\left.\psi\of\phi\inv\right|_{\phi(U\cap{}V)}:\phi(U\cap{}V)\to\psi(U\cap{}V)
	\]
}

\par\noindent
Note: $\phi(U\cap{}V)$ and $\psi(U\cap{}V)$ are open in $\R^{n}$, because $\phi$ and $\psi$ are homeomorphisms.\n
Note: Transition functions are automatically homeomorphisms.\n

\defn{
	Two charts of a topological manifold are \underline{$C^{\infty}$-compatible} (or just \underline{compatible}) iff their transition functions are $C^{\infty}$. That is,
	\[
		\restr{\psi\of\phi\inv}{\phi(U\cap{}V)}\qquad\ptxt{and}\qquad\restr{\phi\of\psi\inv}{\psi(U\cap{}V)}
	\]
	are both $C^{\infty}$ diffeomorphisms.\n
}

\defn{
	An \u{atlas} of a topological manifold $M$ is a collection $\mathscr{A}=\set{(U_{i},\phi_{i})}_{i\in{}I}$ of charts \st{} $M=\bigcup_{i\in{}I}U_{i}$.\n
}

\par\noindent
Preliminary ``definition'': An atlas of $M$ \st{} $\forall{}i,j\in{}I$, the transition function $\phi_{i}\of\phi_{j}\inv$ is $C^{\infty}$ determines a differentiable structure on $M$. Note that the condition is vacuous if $U_{i}\cap{}U_{j}=\emptyset$.\n

\ex{
	Some topological manifolds and atlases satisfying the preliminary definition:
	\begin{itemize}
		\item A trivial example: $M\subseteq\R^{n}$ any open set, $\mathscr{A}=\set{M\hookrightarrow\R^{n}\ptxt{ (inclusion)}}$.
		\item Let $A\subseteq\R^{n}$ be an open set, and $G:A\to\R^{k}$ a $C^{\infty}$ map. Let $M$ be the graph of $G$, i.e.,\break $M=\set{(x,G(x))\in\R^{n+k}\mid{}x\in{}A}\subseteq\R^{n+k}$ with the subspace topology. Then let $\mathscr{A}=\set{\pi:M\to\R^{n}\mid{}\pi:(x,G(x))\mapsto{}x}$.
		\item Cases of $M\subseteq\R^{N}$ which are \u{locally} graphs. (Note: $\R^{N}$ is known as the ``ambient space''.)
		\begin{itemize}
			\item $S^{1}$. Let
			\begin{align*}
				U_{1} & =\set{(x,\sqrt{1-x^{2}});x\in(-1,1)}\\
				U_{2} & =\set{(y,\sqrt{1-y^{2}});y\in(-1,1)}\\
				U_{3} & =\set{(x,-\sqrt{1-x^{2}});x\in(-1,1)}\\
				U_{4} & =\set{(y,-\sqrt{1-y^{2}});y\in(-1,1)}\\
				\mathscr{A} & =\set{(U_{1},(x,y)\mapsto{}x),(U_{2},(x,y)\mapsto{}y),(U_{3},(x,y)\mapsto{}x),(U_{4},(x,y)\mapsto{}y)}
			\end{align*}
			Let's explicitly compute a transition map. $\phi_{1}\inv(x)=(x,\sqrt{1-x^{2}})$, so $\phi_{2}\of\phi_{1}\inv(x)=\sqrt{1-x^{2}}$. Note: this is $C^{\infty}$ on $(0,1)$.
			\item $S^{1}$ with a new atlas. Let $p=(u,v)$. Let $U_{+}=\set{S^{1}\cut\set{(0,1)}}$ and $U_{-}=\set{S^{1}\cut\set{(1,0)}}$.\n
			Then let $\phi_{+}(p)=x=\frac{u}{1-v}$ and $\phi_{-}(p)=y=\frac{u}{1+v}$. Another atlas: $\mathscr{B}=\set{(U_{1},\phi_{1}),(U_{2},\phi_{2})}$. We claim that $\phi_{1}$ and $\phi_{2}$ are $C^{\infty}$-compatible.\n
			In fact, it turns out that $\mathscr{A}\cup\mathscr{B}$ consists of compatible charts. So $\mathscr{A}$ and $\mathscr{B}$ define the same differentiable structure on $S^{1}$.
		\end{itemize}
	\end{itemize}
}
