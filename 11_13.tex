	\item Review differential forms
	\item Pullbacks
	\item Exterior derivatives
\end{itemize}\up\n

\par\noindent
Let $M$ be a manifold. Last time, we defined a smooth $k$-form $\alpha$ on $M$ as an assignment $M\ni{}p\mapsto\alpha_{p}\in\bigwedge^{k}T^{*}_{p}M$.\n

\par\noindent
In local coordinates $(x^{1},\ldots,x^{n})$, $\alpha={\sum_{I}}'a_{I}dx^{I}$, $I=\set{i_{1}<\cdots<i_{k}}\subseteq\set{1,\ldots,n}$.\n

\par\noindent
Smoothness: $\forall{}I$, for any coordinate chart, the $a_{I}$ are $C^{\infty}$.\n

\defn{
	$\Omega^{k}(M)\eqdef\set{\ptxt{all $C^{\infty}$ $k$-forms}}$.\n
}

\ex{
	On $\R^{n}$: volume form $dx^{1}\wedge\cdots\wedge{}dx^{n}$.\n
	On $\R^{3}$: $\Omega^{1}=\set{\alpha=fdx+gdy+hdz}$.\n
	\phantom{On $\R^{3}$: }$\Omega^{2}=\set{\alpha=fdx\wedge{}dy+gdy\wedge{}dz+hdx\wedge{}dz}$.\n
}

\defn{
	Take $M\subseteq\R^{3}$ a surface such that ther is a smooth unit normal vector field $\vec{n}$ on $M$. Define a $2$-form $\sigma$ on $M$ by $\forall{}p\in{}M$, $v,w\in{}T_{p}M\subset\R^{3}$, $\sigma_{p}(v,w)=\det(v,w,\vec{n}_{p})$. So $\sigma_{p}(v,w)$ is the area of the parallelogram spanned by $v$, $w$, and $\vec{n}_{p}$. $\sigma$ is called the \u{area form} of $M$, for the given $\vec{n}$ (orientation).\n
}

\subsection*{Pull-backs of Differential Forms}

\defn{
	Let $F:N\to{}M$ be smooth, and $\alpha\in\Omega^{k}(M)$, We define the \u{pullback of $\alpha$ by $F$}, $(F^{*}\alpha)\in\Omega^{k}(N)$, by $\forall{}p\in{}N$, $v_{1},\ldots,v_{k}\in{}T_{p}N$,
	\[
		(F^{*}\alpha)_{p}(v_{1},\ldots,v_{k})=\alpha_{F(p)}(F_{*,p}(v_{1}),\ldots,F_{*,p}(v_{k}))
	\]
}

\lemma{
	\begin{enumerate}[leftmargin=4\parindent, topsep=0pt, itemsep=0pt]
		\item $F^{*}\alpha$ is $C^{\infty}$.
		\item $(F\of{}G)^{*}\alpha=G^{*}(F^{*}\alpha)$ (the chain rule).
		\item $F^{*}(\alpha\wedge\beta)=(F^{*}\alpha)\wedge(F^{*}\beta)$.
	\end{enumerate}\up\n
}

\par\noindent
Observe: $\Omega^{0}(M)=C^{\infty}(M)$. If $f$ is a $0$-form on $M$, then $F^{*}f=f\of{}F$.\n

\subsubsection*{Pullbacks in Coordinates}

\par\noindent
Given
\[
	\begin{tikzcd}
		\R^{n} &[25pt] V\subset{}N \arrow[l, "(y^{1}{,}\ldots{,}y^{n})", swap] \arrow[r, "F"] & M\supset{}U \arrow[r, "(x^{1}{,}\ldots{,}x^{m})"] &[25pt] \R^{m}
	\end{tikzcd}
\]
and $\alpha={\sum_{I}}'a_{I}dx^{I}$ with the $a_{I}\in{}C^{\infty}(U)$, then
\[
	F^{*}(\alpha)={\sum_{I}}'(a_{I}\of{}F)F^{*}(dx^{I})={\sum_{I}}'(a_{I}\of{}F)F^{*}(dx^{i_{1}}\wedge\cdots\wedge{}dx^{i_{k}})={\sum_{I}}'(a_{I}\of{}F)(F^{*}(dx^{i_{1}})\wedge\cdots\wedge{}F^{*}(dx^{i_{k}}))
\]

\newpage
\lemma{
	Let $F^{i}=x^{i}\of{}F:V\to\R$ for each $i$. Then $F^{*}(dx^{i})=dF^{i}$, the differential of $F^{i}$.\nn
	Proof: First, introduce some shorthand notation: $\d_{y^{j}}=\pd{}{y^{j}}$. Now,
	\[
		F^{*}(dx^{i})(\d_{y^{j}})=dx^{i}(F_{*}\d_{y^{j}})\labeledeq{(1)}dx^{i}\sum_{l=1}^{m}\pd{F^{l}}{y^{j}}\d_{x^{l}}=F^{*}(dx^{i})(\d_{y^{j}})=\pd{F^{i}}{y^{j}}
	\]
	with (1) because $F'=\begin{pmatrix}\pd{F^{l}}{y^{j}}\end{pmatrix}_{(j,l)}$ is the matrix of $F_{*}$ in $(\d_{y^{j}}),(\d_{x^{i}})$. Thus,
	\[
		F^{*}(dx^{i})=\sum_{j=1}^{n}\pd{F^{i}}{y^{j}}dy^{j}=dF^{i}
	\]
	\proven
}

\par\noindent
Now, back to the main computation:
\[
	F^{*}(\alpha)=\cdots={\sum_{I}}'(a_{I}\of{}F)(dF^{i_{1}}\wedge\cdots\wedge{}dF^{i_{k}})
\]
Observe: The right hand side is a smooth form. There's a special case for $k=m=n$:\n

\prop{
	$F^{*}(dx^{1}\wedge\cdots\wedge{}dx^{n})=\det\underbrace{\begin{pmatrix}\pd{F^{i}}{y^{j}}\end{pmatrix}_{(j,i)}}_{\mathclap{=F'=J(F)}}(dy^{1}\wedge\cdots\wedge{}dy^{n})$\nn
	Proof: Well, the left hand side is
	\[
		dF^{1}\wedge\cdots\wedge{}dF^{n}=\underbrace{\paren{\sum_{j_{1}=1}^{n}\pd{F^{1}}{y^{j_{1}}}dy^{j_{1}}}}_{dF^{1}}\wedge\cdots\wedge\underbrace{\paren{\sum_{j_{n}=1}^{n}\pd{F^{n}}{y^{j_{n}}}dy^{j_{n}}}}_{dF^{n}}=\sum_{\hspace{-3em}\mathrlap{j_{1},\ldots,j_{n}=1}}^{n}\;\;\;\;\paren{\prod_{i=1}^{n}\pd{F^{i}}{y^{j_{i}}}}(dy^{j_{1}}\wedge\cdots\wedge{}dy^{j_{n}})
	\]
	Observe that the terms of the sum with $j_{a}=j_{b}$ with $a\ne{}b$ vanish, so the sum is really over all orderings of $\set{1,\ldots,n}$.
	\[
		dF^{1}\wedge\cdots\wedge{}dF^{n}=\sum_{\sigma\in{}S_{n}}\paren{\prod_{i=1}^{n}\pd{F^{i}}{y^{\sigma(i)}}}\underbrace{\paren{dy^{\sigma(1)},\ldots,dy^{\sigma(n)}}}_{=\underbrace{(-1)^{\sigma}}_{\sgn(\sigma)}dy^{1}\wedge\cdots\wedge{}dy^{n}}=\underbrace{\paren{\sum_{\sigma\in{}S_{n}}(-1)^{\sigma}\prod_{i=1}^{n}\pd{F^{i}}{y^{\sigma(i)}}}}_{\det{}F'}\paren{dy^{1}\wedge\cdots\wedge{}dy^{n}}
	\]
	\proven
}

\cor{
	If $\alpha=fdx^{1}\wedge\cdots\wedge{}dx^{n}$ on $\R^{n}$, $f\in{}C^{\infty}_{0}(U)$ (i.e. $f$ has compact support), and we define
	\[
		\int\alpha=\underbrace{\int{}fdx^{1}\wedge\cdots\wedge{}dx^{n}}_{\ptxt{Riemann Integral}}
	\]
	Then $\int{}F^{*}\alpha=\int\alpha$ by the change of variables formula, provided that $\det(F')>0$.\nn
	Proof: $F^{*}\alpha=(f\of{}F)\det(F')dy^{1}\wedge\cdots\wedge{}dy^{n}$.\proven
}

\newpage
\subsection*{The Exterior Differential}

\thm{
	Let $M$ be a manifold. Then there exists a unique operator $d$
	\[
		\Omega^{0}\overset{d}{\xrightarrow{\hspace{0.5cm}}}\Omega^{1}\overset{d}{\xrightarrow{\hspace{0.5cm}}}\Omega^{2}\overset{d}{\xrightarrow{\hspace{0.5cm}}}\cdots\overset{d}{\xrightarrow{\hspace{0.5cm}}}\Omega^{k-1}\overset{d}{\xrightarrow{\hspace{0.5cm}}}\Omega^{k}
	\]
	\st{}
	\begin{enumerate}[itemsep=0pt, topsep=0pt, leftmargin=4\parindent, label=(\arabic*)]
		\item $d$ is $\R$-linear.
		\item $d:\Omega^{0}\to\Omega^{1}$ is the usual differential of a function.
		\item $d^{2}=0$.
		\item The anti-derivation property: $d(\alpha\wedge\beta)=d\alpha\wedge\beta+(-1)^{\deg\alpha}\alpha\wedge{}d\beta$. (This is also known as the super-symmetric version of Leibniz' rule.)
		\item If $F:M\to{}N$ is smooth, then $\forall\alpha\in\Omega^{k}N$, $d(F^{*}\alpha)=F^{*}(d\alpha)$. (This property is called ``Naturality''.)
