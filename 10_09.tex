\par\noindent
Recall that a submersion $F:M\to{}N$ satisfies $\forall{}p\in{}M$, $F_{*,p}$ is onto.\n

\par\noindent
Fibrations are a special class of submersions. For any submersion, its fibers are $F\inv(q)$, $\forall{}q\in{}N$. $F\inv(q)$ is a regular submanifold by the regular value theorem. But not all fibers are diffeomorphic to each other. And $F\inv(q)=\emptyset$ is possible.\n

\par\noindent
Also, the normal form theorem for submersions (also called ``the submersion theorem'') states that locally, a submersion is a map of the form $(x',x'')\mapsto{}x'$.\n

\subsection*{Immersions}

\defn{
	A smooth map $F:M\to{}N$ is an \u{immersion at $p\in{}M$} $\Leftrightarrow$ $F_{*,p}$ is injective. Note that this requires $\dim{}M\le\dim{}N$.\n
}

\defn{
	$F$ is an \u{immersion} iff $\forall{}p\in{}M$, $F$ is an immersion at $p$.\n
}

\ex{
	Our model case is $\R^{k}\hookrightarrow\R^{n}$ with $r\mapsto(r,0)$, where $k\le{}n$.\n
}

\ex{
	\begin{enumerate}[label=\arabic*), leftmargin=3\parindent]
		\item If $S\subset{}N$ is a regular submanifold, the inclusion $\iota:S\hookrightarrow{}N$ is an immersion, since $\forall{}p\in{}S$, $\iota_{*,p}:T_{p}S\hookrightarrow{}T_{p}N$.
		\item Immersions don't have to be injecive. Consider the lemniscate $F:\R\to\R^{2}$. $\forall{}t\in\R$, $\dot{F}(t)\ne{}0$, so $F$ is an immersion. But it's not injective!
		\item Restrict the domain of the lemniscate to obtain a one-to-one map whose image is a regular submanifold.
		\item Can also restrict the domain of the lemniscate to obtain a one-to-one map whose image is \u{not} a regular submanifold. (I.e. it goes right up to the point where the curve would self intersect, but doesn't map to the intersection point more than once.)
		\item Let $N=\R^{2}/\Z^{2}$ (where $\Z^{2}$ is the integer lattice). Define $\pi:\R^{2}\to{}N$, and declare this to be a local diffeomorphism, giving us a smooth structure on $N$. This is the $2$-Torus. Let $F:\R\to\R^{2}\overset{\pi}{\to}N$. Fix $v=(a,b)\in\R^{2}\cut\set{0}$. Sy $F(t)=\pi(tv)$, $\forall{}t\in\R$. If $a$ and $b$ are rational, you get a periodic curve around the torus. These are all immersions. If $a/b$ is irrational, then the image of $F$ in $\R^{2}/\Z^{2}$ is dense! With the subspace topology, $F(\R)$ is very much so \i{not} locally Euclidean.
	\end{enumerate}\up\n
}

\thm{
	(Normal Form for Immersions) Let $F:M\to{}N$ be an immersion at $p\in{}M$. Then there are coordinates $p\in{}U,\phi=(x^{1},\ldots,x^{m})$ of $M$ and $f(p)\in{}V,\psi=(y^{1},\ldots,y^{n})$ of $N$ such that $U\subset{}F\inv(V)$ and $\tilde{F}(r^{1},\ldots,r^{m})=(r^{1},\ldots,r^{n},\underbrace{0,\ldots,0}_{n-m\ptxt{ zeros}})$. We have
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			M\supseteq{}U \arrow[r, "F"] \arrow[d, "\phi"] \& V\subseteq{}N \arrow[d,"\psi"]\\
			\R^{m}\supseteq\phi(U) \arrow[r,"\tilde{F}"] \& \psi(V)\subseteq{}\R^{n}
		\end{tikzcd}
	\]

}

\defn{
	An immersion $F:M\to{}N$ hich is a homeomorphism onto $F(M)$ with the subspace topology is called an \u{embedding}. This is a global property!\n
}

\par\noindent
Observe: Embeddings are injective.\n
