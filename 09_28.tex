\par\noindent
Recall: For $F:M\to{}N$ $C^{\infty}$, and $p\in{}M$, we have
\[
	\map{F_{p}^{*}:C^{\infty}_{F(p)}(N)}{C_{p}^{\infty}(M)}{[f]}{[f\of{}F]}
\]
By duality, we get $F_{*,p}:T_{p}M\to{}T_{F(p)}N$ defined so that, for $v\in{}T_{p}M$, $F_{*,p}(v)[f]=v[f\of{}F]$. This is also called the differential.\n

\par\noindent
$F^{*}_{p}$ is a ring morphism, which maps the ideal
\[
	I_{F(p)}=\set{[f]\in{}C^{\infty}_{p}\mid{}f(F(p))=0}
\]
into
\[
	I_{p}=\set{[f]\in{}C^{\infty}_{p}\mid{}f(p)=0}
\]
This induces a map
\[
	\begin{tikzcd}
		I_{F(p)}/I_{F(p)}^{2} \arrow[r,"F_{p}^{*}"] \arrow[d,symbol=\cong] & I_{p}/I_{p}^{2} \arrow[d,symbol=\cong]\\
		T_{F(p)}^{*} \arrow[r,"F_{p}^{*}"] & T_{p}^{*}M
	\end{tikzcd}
\]
Check that $F_{p}^{*}$ is dual to $F_{p,*}$.\n

\thm{
	(Chain Rule) Let $M\overset{F}{\to}N\overset{G}{\to}O$ be smooth, and $p\in{}M$. Then $(G\of{}F)_{*,p}=G_{*,F(p)}\of{}F_{*,p}$.\nn
	Proof: Let $[f]\in{}C^{\infty}_{G(F(p))}(O)$. Then $f\of(G\of{}F)=(f\of{}G)\of{}F$. Now pick $v\in{}T_{p}M$. Then
	\[
		(G\of{}F)_{*,p}(v)[f]=v(f\of(G\of{}F))=v((f\of{}G)\of{}F)=F_{*,p}(v)(f\of{}G)=G_{*,F(p)}(F_{*,p}(v))[f]
	\]
	So $(G\of{}F)_{*,p}=G_{*,F(p)}\of{}F_{*,p}$.\proven
}

\ex{
	Let $p\in{}U\osubseteq{}M$, $\phi:U\to\R^{n}$ a coordinate chart. As usual, write $\phi=(x^{1},\ldots,x^{n})$, with $x^{i}:U\to\R$. Say $\R^{n}=N$, a manifold with a single chart, the identity map $r=(r^{1},\ldots,r^{n})$, $r^{i}:\R^{n}\to\R$.\n
	Claim: $\phi_{*,p}(\pdat{}{x^{i}}{p})=\pdat{}{r^{i}}{\phi(p)}$. In other words, partial derivatives in $\R^{n}$ correspond with standard basis vectors of $T_{p}M$, via the pushforward.\nn
	Proof: We can form $f_{\phi}=f\of\phi\inv$. So by definition, $(\phi\inv)_{*,\phi(p)}(\pd{}{r^{i}})=\pd{}{x^{i}}$. By the chain ule, $(\phi\inv)_{*,\phi(p)}=(\phi_{*,p})\inv$.\proven
}

\subsection*{Differentials of Functions}

\par\noindent Let $f:M\to{}N=\R$ (with a single chart, the identity map). Note that $\forall{}a\in\R$, we can identify $T_{a}\R\cong\R$ using $\pdat{}{r}{a}$ as a basis of $T_{a}\R$.\n

\par\noindent
Claim: Then $f_{*,p}:T_{p}M\to{}T_{f(p)}\R=\R$ is the same as $df_{p}:T_{p}M\to\R$ (defined as the class of $[f-f(p)]\in{}I_{p}$ in the quotient $I_{p}/I_{p}^{2}$).\n

\par\noindent
Chain rule: $M\overset{F}{\to}N\overset{f}{\to}\R$, $v\in{}T_{p}M$ simply reads
\[
	(f\of{}F)_{*,p}(v)=(f_{*,F(p)}\of{}F_{*,p})(v)=df_{p}(F_{*,p}(v))=F_{p}^{*}(df_{p})(v)
\]
by the duality between $F_{*}^{p}$ and $F_{*,p}$.\n

\par\noindent
Conclusion: The pullback map on differentials is the pushforward of the composition, i.e., $F_{p}^{*}(df_{p})=(f\of{}F)_{*,p}=d(f\of{}F)_{p}$. So we frequently write $F_{*,p}=dF_{p}$.

\subsection*{Computation of $F_{*,p}$ in Coordinates}

\par\noindent
Let $F:M\to{}N$, $p\in{}M$. Let $(V,\psi)$ be a coordinate chart near $F(p)$.
\[
	\begin{tikzcd}
		p\in{}F\inv(V)\subseteq{}U\subseteq{}M \arrow[r,"F\,:\,M\to{}N"] \arrow[d,"\phi\,=\,(x^{1}{,}\ldots{,}x^{m})"] &[25pt] F(p)\in{}V\subseteq{}N \arrow[d,"\psi\,=\,(y^{1}{,}\ldots{,}y^{n})"]\\
		\R^{m} \arrow[r, "\tilde F\,=\,\psi\of{}F\of\phi\inv\,:\,\phi(U)\to\R^{n}"] & \R^{n}
	\end{tikzcd}
\]

\lemma{
	The matrix of $F_{*,p}$ in the ordered bases
	\[
		\paren{\pdat{}{x^{1}}{p},\ldots,\pdat{}{x^{m}}{p}}\subseteq{}T_{p}M\qquad\ptxt{and}\qquad\paren{\pdat{}{y^{1}}{F(p)},\ldots,\pdat{}{y^{n}}{F(p)}}\subseteq{}T_{F(p)}N
	\]
	is simply $\begin{pmatrix}\pd{F^{j}}{x^{i}}(p)\end{pmatrix}$, the Jacobian, where $F^{j}=y^{j}\of{}F:U\to\R$, with $\psi\of{}F=(F^{1},\ldots,F^{n})$.\nn
	Observe: This is the Jacobian of $\tilde{F}$ at $\phi(p)$ (in the Calc III sense).\nn
	Proof: We want to compute the component $F_{*,p}(\pd{}{x^{i}})$ with respect to $\pd{}{y^{j}}$. This is
	\[
		F_{*,p}(\pd{}{x^{i}})([y^{j}])=\restr{\pd{}{x^{i}}(y^{j}\of{}F)}{p}=\restr{\pd{}{x^{i}}(F^{j})}{p}
	\]
	\proven
}

\defn{
	A $C^{\infty}$ map $F:M\to{}N$ is a \u{local diffeomorphism} iff $\forall{}p\in{}M$, there are open neighborhoods $U$ of $p$ and $V$ of $F(p)$, such that $F(U)=V$ and $F|_{U}^{V}:U\to{}V$ has a smooth inverse $(F|_{U}^{V})\inv:V\to{}U$.\n
}

\ex{
	$F:S^{n}\to\RP^{n}$ is a local diffeomorphism, but not a global diffeomorphism.\n
}
