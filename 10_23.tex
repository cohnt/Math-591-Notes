\i{Note: The lecture on 10/21 was devoted to review for the exam.}\n

\par\noindent
Recall: We defined, for $X,Y\in\mathfrak{X}(M)$, $[X,Y]$ by regarding $X$ and $Y$ as operators on $C^{\infty}(M)$. Then $[X,Y]$ is the commutator $X\of{}Y-Y\of{}X$.\n

\par\noindent
Properties:
\begin{itemize}[topsep=0pt, itemsep=0pt, label=-]
	\item $[\,\cdot\,,\,\cdot\,]$ is bilinear over $\R$.
	\item $[\,\cdot\,,\,\cdot\,]$ satisfies the Jacobi identity: $\forall{}X,Y,Z\in\mathfrak{X}(M)$, $[[X,Y],Z]+[[Z,X],Y]+[[Y,Z],X]=0$.\n
	Proof: Just compute! It's very straightforward, and only uses the fact that composition is associative.
\end{itemize}\up\n

\defn{
	A \u{Lie algebra} $\mathfrak{g}$ is a vector space (over $\R$), together with a bilinear map $[\,\cdot\,,\,\cdot\,]:\mathfrak{g}\times\mathfrak{g}\to\mathfrak{g}$, which
	\begin{itemize}[topsep=0pt, itemsep=0pt, leftmargin=4\parindent, label=-]
		\item Is skew-symmetric: $\forall{}A,B\in\mathfrak{g}$, $[A,B]=-[B,A]$.
		\item Satisfies the Jacobi identity.
	\end{itemize}\up\n
}

\ex{
	For any manifold $M$, $\mathfrak{g}=\mathfrak{X}(M)$ is a Lie algebra.\n
}

\ex{
	A finite dimensional example: Let $\mathfrak{g}=\set{\ptxt{$n\times{}n$ skew-symmetric real matrices}}\cong{}T_{I}O(n)$, with $[A,B]=AB-BA$ (using matrix multiplication).\n
	Check: $[A,B]\in\mathfrak{g}$. Well, $(AB-BA)\tpose=B\tpose{}A\tpose-A\tpose{}B\tpose{}=(-B)(-A)-(-A)(-B)=BA-AB=-(AB-BA)$.\n
	We say $o(n)\eqdef{}\mathfrak{g}$.\n
}

\par\noindent
At some point (next week), we will prove the following:\n
Claim: Given $A\in{}T_{I}O(n)$, define $A^{\sharp}\in\mathfrak{X}(O(n))$ by $\forall{}g\in{}O(n)$, $A^{\sharp}_{g}=gA$, and we claim that $A^{\sharp}_{g}\in{}T_{g}O(n)$.\n

\par\noindent
To interpret the formula, let $\map{L_{g}:O(n)}{O(n)}{k}{gk}$. Then $A^{\sharp}_{g}=(L_{g})_{*,I}(A)$, i.e., $(L_{g})_{*,I}:T_{I}O(n)\to{}T_{g}O(n)$.\n
Then $\forall{}A,B\in{}T_{I}O(n)$, $\underbrace{[A,B]^{\sharp}}_{\substack{\ptxt{matrix}\\ \ptxt{commutator}}}=\underbrace{[A^{\sharp},B^{\sharp}]}_{\substack{\ptxt{vector field}\\ \ptxt{commutator}}}$. So $\sharp:o(n)\rightarrowtail\mathfrak{X}(O(n))$ is an injection, and $A^{\sharp}_{I}=A$.\n
So $o(n)$ appears as a (finite-dimensional) Lie subalgebra of $\mathfrak{X}(O(n))$.\n
There is more about this to come in the next chapter...\n

\par\noindent
Now, back to the general study of vector fields. We've looked at vector fields as operators. Now, let's look at them as generators of dynamics.\n

\defn{
	Let $X\in\mathfrak{X}(M)$. An \u{integral curve} of $X$ is a map
	\[
		c:(a,b)\to{}M\qquad{}a<b;a,b\in\R\cup\set{\pm\infty}
	\]
	such that $\forall{}t\in(a,b)$ ``time'', $\dot{c}(t)=X_{c(t)}\in{}T_{c(t)}M$.\n
}

\newpage
\thm{
	(Existence and Uniqueness of Integral Curves)\n
	Given $X\in\mathfrak{X}(M)$, $p\in{}M$, then there exists an integral curve of $X$, $c:(a,b)\to{}M$, with $a<0<b$, such that $c(o)=p$. Moreover, any two such curves agree on the intersection of their domains.\nn
	Proof: Reduce to the Euclidean case near $p$. Introduce local coordinates near $p$. WOLOG $\phi(p)=0\in\R^{n}$. We can write
	\[
		\restr{X}{U}=\sum_{i=1}^{n}f_{i}\pd{}{x^{i}}
	\]
	where the $f_{i}$ are smooth functions on $U$. Then we have
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			(a,b) \arrow[r, "c"] \arrow[dr, "\tilde{c}", swap] \& U \arrow[d, "\phi"] \\[-5pt]
			\& \R^{n}
		\end{tikzcd}
	\]
	with the unkown curve $\tilde{c}(t)=(x^{1}(t),\ldots,x^{n}(t))$, with each $x^{i}:(a,b)\to\R$ unknown. Note:
	\[
		\left\{\begin{array}{l}
			\dot{c}(t)=X_{c(t)}\\
			c(0)=p
		\end{array}\right.\qquad\Leftrightarrow\qquad\left\{\begin{array}{l}
			\forall{}i,\;x^{i}(t)=f_{i}(c(t))\\
			x^{i}(0)=0
		\end{array}\right.
	\]
	Well, $f^{i}(c(t))=\tilde{f}_{i}(x^{1}(t),\ldots,x^{n}(t))$, and these are real-valued functions, so $\tilde{x}^{i}(t)=\tilde{f}_{i}(x^{1}(t),\ldots,x^{n}(t))$. So we have a system of ordinary differential equations (ODEs), with an initial condition.\n
	Well, the derivatives $\frac{dx^{i}(t)}{dt}=\dot{x}^{i}(t)$ are the left-hand side, so we can just quote the existence and uniqueness theorems from the theory of ordinary differential equations. Thus, we get the theorem in a coordinate chart. Now, consider a covering of $c$ by overlapping coordinate charts...\proven
}

\ex{
	Let $M=(\R_{>0})^{2}$ -- the upper-left quadrant of $\R^{2}$, with standard coordinates $(x,y)$.\n
	Let $X=yx^{2}\pd{}{x}$.\nn
	Well, $\dot{x}=yx^{2}$ and $\dot{y}=0$, so $\forall{}t$, $y(t)=y(0)$. So $\frac{dx}{dt}=y(0)x^{2}$. We can solve this using separation of variables:
	\begin{align*}
		\frac{dx}{x^{2}} & =y(0)dt\\
		\int{}x^{-2}\,dx & =\int{}y(0)\,dy\\
		-\frac{1}{x(t)}+\frac{1}{x(0)} & =y(0)t\\
		\frac{1}{x(t)} & =\frac{1}{x(0)}-yt=\frac{1-y(0)x(0)t}{x(0)}\\
		x(t) & =\frac{x(0)}{1-x(0)y(0)t}
	\end{align*}
	And we also have $y(t)=y(0)$. Well, $x$ will escape to infinity at the ``escape time'' $T=\frac{1}{x(0)y(0)}$. So the maximal domain of an integral curve with these initial conditions is $(-\infty,T)$.\n
}

\par\noindent
This demonstrates a problem -- if we want to look at maximal integral curves, the maximal domain may depend on the initial conditions.\n
