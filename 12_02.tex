\par\noindent
Continuing from last time, W.T.S. $\displaystyle\sum_{\alpha}\int(\phi_{\alpha}\inv)^{*}(\chi_{\alpha}\mu)$ is independent of choice of coordinates and partition of unity. Say $\set{\tilde\phi_{\beta}},\set{\tilde\chi_{\beta}}$ is another choice. Then $\forall\alpha$,
\[
	\chi_{\alpha}\mu=\sum_{\beta}\tilde\chi_{\beta}\chi_{\alpha}\mu\qquad\sum_{\beta}\tilde\chi_{\beta}=1
\]
Thus,
\[
	\int(\phi_{\alpha}\inv)^{*}(\chi_{\alpha}\mu)=\sum_{\beta}\int(\phi_{\alpha}\inv)^{*}(\underbrace{\tilde\chi_{\beta}\chi_{\alpha}\mu}_{\hspace{-30pt}\mathrlap{\supp\ptxt{ in }(\dom\tilde\phi_{\beta})\cap(\dom\phi_{\alpha})}})
\]
So $\forall\beta$,
\[
	\int(\phi_{\alpha}\inv)^{*}(\tilde\chi_{\beta}\chi_{\alpha}\mu)=\int(\tilde\phi_{\beta}\inv)^{*}(\tilde\chi_{\beta}\chi_{\alpha}\mu)
\]
because of the invariance of integrals of top-degree forms in Euclidean space, under orientation-preserving diffeomorphisms. (In this case, the transition function is that orientation-preserving diffeomorphism.) Now, back to the full term in the integral:
\[
	\sum_{\alpha}(\phi_{\alpha}\inv)^{*}(\chi_{\alpha}\mu)=\sum_{\alpha,\beta}(\tilde\phi_{\beta}\inv)^{*}(\tilde\chi_{\beta}\chi_{\alpha}\mu)
\]
$\forall\beta$, sum over $\alpha$ first, and use the fact that $\sum_{\alpha}\chi_{\alpha}=1$. This yields
\[
	\sum_{\alpha}(\phi_{\alpha}\inv)^{*}(\chi_{\alpha}\mu)=\cdots=\sum_{\beta}(\tilde\phi_{\beta}\inv)^{*}(\tilde\chi_{\beta}\mu)
\]
\proven

\par\noindent
Observe: In practice, don't use partitions of unity. Use parameterizations that partition $M$ (or at least, $\supp\mu\subset{}M$), up to sets of measure $0$.\n

\ex{
	On a torus, pullback $\mu$ to $\R^{n}$ and integrate over a fundamental domain.\n
}

\defn{
	Let $S\subset{}M$ be an oriented $k$-dimensional submanifold. Define
	\[
		\map{\displaystyle\int_{S}:\Omega_{0}^{k}(M)}{\R}{\alpha}{\displaystyle\int_{S}\iota^{*}\alpha}
	\]
	where $\iota:S\hookrightarrow{}M$ is the inclusion map. In general, we omit the $\iota^{*}$, and just say
	\[
		\int_{S}\alpha\eqdef\int_{S}\iota^{*}\alpha
	\]
	\n
}

\subsection*{Manifolds with Boundary}

\par\noindent
These are needed for Stokes' theorem. We begin with some preliminary definitions...\n

\defn{
	Let $S,T\subset\R^{n}$ (with no assumptions of their properties). $S$ and $T$ are \u{diffeomorphic} iff $\exists{}U,V\osubset\R^{n}$ \st{} $S\subset{}U$, $T\subset{}V$, and $\exists{}F:U\to{}V$ a diffeomorphism \st{} $F(S)=T$. We will say $\restr{F}{S}^{T}:S\to{}T$ is a \u{diffeomorphism} between the sets.\n
}

\lemma{
	If $U\subset\R^{n}$ is open and diffeomorphic to $T\subset\R^{n}$, then $T$ is open.\nn
	Proof: By definition, $\exists{}V,\tilde{U}$ open with $T\subset{}V$ and $U\subset\tilde{U}$, and $\exists{}F:\tilde{U}\to{}V$ diffeomorphic such that $F(U)=T$. Then $\restr{F}{U}:U\to\R^{n}$ is an open map, so $T$ is open.\proven
}

\defn{
	$\H^{n}=\set{(x^{1},\ldots,x^{n})\in\R^{n}\mid{}x_{n}\ge{}0}$.\n
}

\ex{
	$B_{r}(0)\cap\H^{n}$. A diffeomorphism on this set must map interior points to interior points, so it maps $B_{r}(0)\cap\set{x^{n}=0}$ to the boundary of its image.\n
}

\par\noindent
In fact, the same works for any relatively open sets of $\H^{n}$! And such diffeomorphisms restrict to diffeomorphisms of $W\cap\d\H^{n}$ onto $F(W)\cap\d\H^{n}$ (for any $W$ relatively open).\n

\defn{
	A \u{topological manifold with boundary} $M$ is a second-countable, Hausdorff topological space that is local homeomorphic to relatively open sets in $\H^{n}$.\n
}

\par\noindent
Note: This includes ordinary open sets in $\R^{n}$, and the sets with boundary.\n

\defn{
	A \u{$C^{\infty}$ atlas} on a topological manifold with boundary is an atlas such that all transition functions are $C^{\infty}$.\n
}

\defn{
	A \u{$C^{\infty}$ manifold with boundary} is a topological manifold with boundary, together with a maximal $C^{\infty}$ atlas.\n
}

\defn{
	Let $M$ be a smooth manifold with boundary. Then the \u{boundary} $\d{}M$ is the set of points $p\in{}M$ which have a nearby chart $\phi$ such that $\phi(p)$ is in the boundary of $\H^{n}$.\n
}

\par\noindent
Note that this is true for one coordinate chart iff it's true for all coordinate charts.\n

\lemma{
	$\d{}M$ inherits a $C^{\infty}$ manifold structure (without boundary) by restricting charts of $M$ defined near boundary points.\n
}

\lemma{
	$\d(\d{}M)=\emptyset$.\n
}

\subsection*{Tangent Spaces}

\par\noindent
They're defined as before!\n

\defn{
	Given $S\subset\R^{n}$, $f:S\to\R^{k}$ is \u{smooth} iff $\exists{}U\subset\R^{n}$ open with $S\subset{}U$, and $\exists{}\tilde{f}:U\to\R^{k}$ smooth such that $\restr{\tilde{f}}{S}\equiv{}f$. We call $\tilde{f}$ a \u{smooth extension}.\n
}

\par\noindent
So, for $f$ smooth, $f\of\phi\inv:W\to\R$, $W\subset\H^{n}$ must be $C^{\infty}$ for all charts $(\phi\inv(W),\phi)$. This gives us germs: $\forall{}p\in{}M$, $C^{\infty}_{p}(M)$. We define $T_{p}M$ to be the set of all derivations on $C^{\infty}_{p}(M)$.\n

\par\noindent
\i{But what happens at the boundary?} If $p\in\d{}M$, what is $T_{p}M$?\n

\par\noindent
Claim: Well, if $p\in\d\H^{n}$, $T_{p}\H^{n}$ is still spanned by $(\pd{}{x^{1}},\ldots,\pd{}{x^{n}})$.\n

\par\noindent
The question is how do we define $\pd{}{x^{n}}$, and how does it act? If $f\in{}C^{\infty}$ near $p$, then $\exists\tilde{f}$, an extension of $f$ to an open set of $\R^{n}$. Then define $\pd{f}{x^{n}}(p)\eqdef\pd{\tilde{f}}{x^{n}}(p)$.\n

\par\noindent
Claim: The RHS is independent of choice of extension. Well,
\[
	\pd{\tilde{f}}{x^{n}}(p)=\lim_{h\to\infty}\frac{\tilde{f}(0,\ldots,h)-\tilde{f}(0,\ldots,0)}{h}
\]
This limit will always be the same, since the limit always exists (by definition of the extension).
