\thm{
	If $M$ is compact, then every $X\in\mathfrak{X}(M)$ is complete.\nn
	Proof: We know (by the existence of local flows) that $\forall{}p\in{}M$, there exists a neighborhood $V$ of $p$, with $\varepsilon_{p}>0$, such that the flow of $X$ is defined on $(-\varepsilon_{p},\varepsilon_{p})\times{}V_{p}\to{}M$. We can extract a finite subcover of $\set{V_{p}\mid{}p\in{}M}$, say $\set{V_{p_{1}},\ldots,V_{p_{k}}}$. Let $\varepsilon=\min\set{\varepsilon_{p_{1}},\ldots,\varepsilon_{p_{k}}}$. Then the flow of $X$ is defined on $(-\varepsilon,\varepsilon)\times{}M$, $\forall{}p$. So by the uniform time lemma, $X$ is complete.\proven
}

\par\noindent
Question: How do vector fields relate with smooth maps? The answer is ``not well''.\n

\par\noindent
In general, vector fields cannot be pushed forward or pulled back. Let $F:M\to{}N$ smooth. We can certainly push forward single tangent vectors: $F_{*,p}:T_{p}M\to{}T_{F(p)}N$. But $F$ may not be injective or surjective. If $F$ is not injective, we have $F(p_{1})=F(p_{2})$ for some $p_{1},p_{2}\in{}M$. If $F$ is not injective, there's a $q\in{}N$ such that $F(p)\ne{}q$, $\forall{}p\in{}M$. In either case, it's not clear what the vector field should be at that point.\n

\par\noindent
Observe: If $F$ is a diffeomorphism, thne given $X\in\mathfrak{X}(M)$, we \i{can} define $(F_{*}X)_{q}$ by $\forall{}q\in{}N$, $(F_{*}X)_{q}=F_{*,F\inv(q)}(X_{F\inv(q)})$.\n

\defn{
	Given $F:M\to{}N$, $X\in\mathfrak{X}(M)$, $Y\in\mathfrak{X}(N)$, we say $X$ and $Y$ are \u{$F$-related} iff $\forall{}p\in{}M$, $F_{*,p}(X_{p})=Y_{F(p)}$.\n
}

\ex{
	Let $F:\R^{2}\to\R$, $F(x,y)=x$. Take $X\in\mathfrak{X}(\R^{2})$ to be $X=f\pd{}{x}+g\pd{}{y}$, $f,g\in{}C^{\infty}(\R^{2})$. When is $X$ $F$-related to some $Y\in\mathfrak{X}(\R)$?\nn
	Well, $F_{*,(x,y)}(X_{(x,y)})=f(x,y)\pd{}{x}$. So only when $f$ doesn't depend on $y$.\n
}

\prop{
	Let $F:M\to{}N$ be smooth, $X\in\mathfrak{X}(M)$, $Y\in\mathfrak{X}(N)$. $X$ and $Y$ are $F$-related iff (1) $\forall{}c$, an integral curve of $X$, $F\of{}c$ is an integral curve of $Y$ (if there are no domain issues) iff (2) the following diagram commutes:
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			C^{\infty}(M) \arrow[d,"X"] \& C^{\infty}(N) \arrow[l, "F^{*}"] \arrow[d,"Y"]\\
			C^{\infty}(M) \& C^{\infty}(N) \arrow[l,"F^{*}"]
		\end{tikzcd}
	\]
	where $F^{*}(g)=g\of{}F$, i.e., $\forall{}g\in{}C^{\infty}(N)$, $X(g\of{}F)=Y(g)\of{}F$.\nn
	Proof: (1) essentially follows directly from the definition, and uniqueness of integral curves.\nn
	For (2), recall how fields act as operators.
	\[
		X(g\of{}F)(p)=d(g\of{}F)(X_{p})=dg(F_{*,p}(X_{p}))
	\]
	\[
		(Y(g)\of{}F)(p)=Y(g)(F(p))=dg(Y_{F(p)})
	\]
	These are equal $\forall{}g$ iff $F_{*,p}(X_{p})=Y_{F(p)}$, which is precisely the condition that $X$ and $Y$ are $F$-related.\proven
}

\prop{
	Given $F:M\to{}N$ smooth, $X_{1},X_{2}\in\mathfrak{X}(M)$, $Y_{1},Y_{2}\in\mathfrak{X}(N)$, if $X_{1}$ is $F$-related to $Y_{1}$ and $X_{2}$ is $F$-related to $Y_{2}$, then $[X_{1},X_{2}]$ is $F$-related to $[Y_{1},Y_{2}]$.\nn
	Proof: Left as an exercise. Use condition (2) from the previous proposition.\n
}

\cor{
	If $X_{1},X_{2}\in\mathfrak{X}(G)$, $G$ a Lie group, and $X_{1}$ and $X_{2}$ are left-invariant, then $[X_{1},X_{2}]$ is left-invariant.\nn
	Proof: $\forall{}g\in{}G$, we know $X_{1}$ and $X_{2}$ are $L_{g}$-related to themselves. Therefore, $[X_{1},X_{2}]$ is $L_{g}$-related to itself, i.e., $[X_{1},X_{2}]$ is left-invariant.\proven
}

\par\noindent
Question: Given $X,Y\in\mathfrak{X}(M)$, we defined $[X,Y]$ regarding $X$ and $Y$ as operators. What is the dynamical interpretation/meaning of the commutator?\n

\thm{
	Let $p\in{}M$, let $\phi$ be the flow of $X$. Form the curve in $T_{p}M$:
	\[
		(-\varepsilon,\varepsilon)\ni{}t\mapsto(d(\phi_{t})_{p})\inv(Y_{\phi_{t}(p)})\eqdef{}v_{t}
	\]
	(with $d(\phi_{t})_{p}:T_{p}M\to{}T_{\phi_{t}(p)}M$). Then $\frac{d}{dt}v_{t}=d(\phi_{t})\inv_{p}([X,Y]_{\phi_{t}(p)})$ (the derivative of a curve in a vector space). At $t=0$, $\frac{d}{dt}v_{t}=[X,Y]_{p}$.\nn
	Proof: Next time.\n
}

\cor{
	If $[X,Y]\equiv0$ (everywhere), then $\forall{}t,s$, $\phi_{t}\of\phi_{s}=\phi_{s}\of\phi_{t}$, where $\phi$ is the flow of $X$ and $\psi$ is the flow of $Y$.
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			p \arrow[d, "\psi"] \arrow[r, "\phi"] \& \phi_{t}(p) \arrow[d, "\psi"]\\
			\psi_{s}(p) \arrow[r] \& \phi_{t}(\psi_{s}(p)) \arrow[r, symbol={=}] \&[-15pt] \psi_{s}(\phi_{t}(p))
		\end{tikzcd}
	\]
	\nn
	Proof: The assumption $[X,Y]=0$ implies the curves $v_{t}$ from above are constant. So $\forall{}t$, $d(\phi_{t})\inv(Y_{\phi_{t}(p)})=Y_{p}$. Thus, $Y$ is $\phi_{t}$-related to itself, $\forall{}t$. So $\phi_{t}$ maps integral curves of $Y$ to integral curves of $Y$. This is equivalent to the commutativity we're trying to show.\proven
}
