\par\noindent
Many examples of $C^{\infty}$ manifolds are produced by the implicit function theorem. Reminder:\n

\thm{
	(Implicit Function Theorem) (Theorem B2 in the textbook)\n
	Let $U\subseteq\R^{N}$ open, $F:U\to\R^{k}$ $C^{\infty}$, and $x_{0}\in{}U$ \st{} $F(x_{0})=0$.\n
	Split: for $x\in{}U$, write $x=(x',x'')$, where $x'\in\R^{N-k}$ and $x''\in\R^{k}$. Accordingly, the Jacobian of $F$ at $x_{0}$ splits:
	\[
		F'(x_{0})=(\underbrace{\tfrac{\partial{}F}{\partial{}x'}(x_{0})}_{N-k}\brokenvert\underbrace{\tfrac{\partial{}F}{\partial{}x''}(x_{0})}_{k})\}k
	\]
	Assume $[\pd{F}{x''}(x_{0})]$ ($k\times{}k$) is invertible. Then there exist open sets $A,B$ with $x_{0}'\in{}A\subseteq\R^{N-k}$, $x_{0}''\in{}B\subseteq\R^{k}$ and $g:A\to{}B$ $C^{\infty}$ \st{} $\set{F\inv(0)}\cap(A\times{}B)=\set{(x',g(x')\mid{}x'\in{}A)}$.\n
}

\par\noindent
Application: Recall that given $F:U\to\R^{k}$, $U\subseteq\R^{N}$ open, zero is a regular value of $F$ iff $\forall{}x\in{}F\inv(0)$, $F'(x)$ has rank $k$.\n

\cor{
	If $0$ is a regular value for $F$, then $F\inv(0)=M$ is locally a graph. Moreover, this structure of local graph gives $M$ a $C^\infty$ atlas, and therefore a smooth manifold structure.\n
	(Note: we will define ``submanifold'', and then $F\inv(0)$ will be examples of submanifolds of $\R^{N}$.)\n
	Proof/Explanation: Assume $0$ is a regular value of $F$. Then $\forall{}x_{0}\in{}F\inv(0)=M$,
	\[
		F'(x_{0})=\begin{pmatrix}
			\horizontalMatrixLine & \del{}f^{1}(x_{0}) & \horizontalMatrixLine\\
			& \vdots &\\
			\horizontalMatrixLine & \del{}f^{k}(x_{0}) & \horizontalMatrixLine
		\end{pmatrix}
	\]
	(for $F=(f^{1},\ldots,f^{k})$). After permuting the indices among the $x_{i}$, without loss of generality $[\pd{F}{x''}(x_{0})]_{k\times{}k}$ is non-degenerate. (Think of this as swapping the columns around so the block is invertible.)\n
}

\par\noindent
The independent variables will depend on $x_{0}$. The number of independent variables is $N-k=\dim{}M$. $k$ is the \u{codimension} of $M\subseteq\R^{N}$.\n

\subsection*{Statement about ``Atlas''}

\par\noindent
Recall: A graph $\set{(x',g(x'))\mid{}x'\in{}A}=\Gamma$ has a global chart: just the projection onto the domain. $\Gamma\ni(x',x'')\mapsto{}x'$. Its inverse is $x'\mapsto(x',g(x'))$.\n

\par\noindent
For two local representations of $M$ as a local graph, transition functions are of the form $x'\mapsto{}(x',g(x'))\overset{\star}{\to}\R^{N-k}$. $\star$ is a projection onto an $N-k$-dimensional coordinate plane of $\R^{N}$. This is smooth, and a transition map.\n

\ex{
	$\frac{x^{2}}{a^{2}}+\frac{y^{2}}{b^{2}}+\frac{z^{2}}{c^{2}}=1$ in $\R^{3}$ is the zero set of $F(x,y,z)=\frac{x^{2}}{a^{2}}+\frac{y^{2}}{b^{2}}+\frac{z^{2}}{c^{2}}-1$.\n
	(Check that $0$ is a regular value.)\n
}

\ex{
	$O(n)=\set{g\in\Mat(n,\R)\mid{}g\inv=g\tpose}$. $\Mat(n,\R)\cong\R^{n^{2}}$, so $O(n)\subseteq\R^{n^{2}}$.\n
	In fact, $O(n)\subseteq\GL(n,\R)\subseteq\R^{n^{2}}$ (and $\GL(n,\R)$ is open in $\R^{n^{2}}$).\n
	Define
	\[
		\map{F:\GL(n,\R)}{\Symm(n,\R)}{g}{gg\tpose-I}
	\]
	where $\Symm(n,\R)=\set{g\in\Mat(n,\R)\mid{}g=g\tpose}\cong\R^{\frac{n(n+1)}{2}}$, and $I$ is the identity matrix. Note: We have to choose the codomain carefully so that $0$ is a regular value.\n
	So, $O(n)=F\inv(0)$.\n
	Check: is $0$ a regular value? To see if $F'(g)$, for $g\in{}O(n)$, has maximal rank, let $M\in\Mat(n,\R)$. Then compute $\left.\frac{d}{dt}F(g+tM)\right|_{t=0}$. (Compute this as a matrix to avoid $\R^{n^{2}}$ coords.) Then $(-\varepsilon,\varepsilon)\ni{}t\mapsto{}\GL(n,\R)\overset{F}{\mapsto}\Symm(n,\R)$.
	\[
		\left.\frac{d}{dt}F(g+tM)\right|_{t=0}=\left.\frac{d}{dt}(g+tM)(g+tM)\tpose\right|_{t=0}=\left.\frac{d}{dt}\paren{gg\tpose+t(Mg\tpose+gM\tpose)+t^{2}MM\tpose}\right|_{t=0}=Mg\tpose+gM\tpose
	\]
	Question: Is $\Mat(n,\R)\ni{}M\mapsto{}Mg\tpose+gM\tpose$ onto? Yes! (This is true iff $F'(g)$ has rank equal to the dimension of $\Symm(n,\R)$. Let $S\in\Symm(n,\R)$. What can $M$ be?)\n
	By the implicit function theorem, $O(n)\subseteq{}\R^{n^{2}}$ is locally a graph. So it has a natural $C^{\infty}$ structure.\n
}
