\subsection*{Tensors}

\defn{
	Let $V$ be a finite-dimensional vector space. A \u{tensor} on $V$ is an element of
	\[
		\underbrace{V^{*}\otimes\cdots\otimes{}V^{*}}_{l}\otimes\underbrace{V\otimes\cdots\otimes{}V}_{m}
	\]
	(where $\otimes$ is the tensor product).\n
}

\par\noindent
Observe: This space is isomorphic to the space of multilinear maps $\underbrace{V\times\cdots\times{}V}_{l}\times\underbrace{V^{*}\times\cdots\times{}V^{*}}_{m}\to\R$, because $(V^{*})^{*}\cong{}V$.\n

\par\noindent
Last time, we defined the set of \i{alternating} multilinear functions $\bigwedge^{k}V^{*}\hookrightarrow\underbrace{V^{*}\otimes\cdots\otimes{}V^{*}}_{k}$.\n

\par\noindent
Reminder: A basis for $\bigwedge^{k}V^{*}$: choose $\set{\mathcal{E}^{i}}_{1\le{}i\le{}n}$ an ordered basis of $V^{*}$. For each $I=\set{i_{1}<\cdots<i_{k}}\subseteq\set{1,\ldots,n}$, let $\mathcal{E}^{I}(v_{1},\ldots,v_{k})=\det(\mathcal{E}^{i_{j}}(v_{l}))_{(j,l)}$.\n

\prop{
	$\set{\mathcal{E}^{I}:I=\set{i_{1}<\cdots<i_{k}}\subseteq\set{1,\ldots,n},\#I=k}$ is a basis of $\bigwedge^{k}V^{*}$.\n
}

\par\noindent
In fact, $\forall\alpha\in\bigwedge^{k}V^{*}$, $\displaystyle\alpha={\sum_{I}}'\alpha(e_{i_{1}},\ldots,e_{i_{k}})\mathcal{E}^{I}$, where $\set{e_{j}}$ is the basis of $V$ dual to $\set{\mathcal{E}^{j}}$ (i.e., $\mathcal{E}^{i}(e_{j})=\delta_{ij}$).\n
Note: the notation $\displaystyle{}{\sum_{I}}'$ means sum over increasing $I=\set{i_{1}<\cdots<i_{k}}$.\n
Also, $\dim\bigwedge^{k}V^{*}=\binom{n}{k}$.\n

\defn{
	By convention, $\bigwedge^{0}V^{*}=\R$.\n
}

\defn{
	If $\alpha\in\bigwedge^{k}V^{*},\beta\in\bigwedge^{l}V^{*}$, we define the \u{wedge product} $\alpha\wedge\beta\in\bigwedge^{k+l}V^{*}$ by
	\[
		(\alpha\wedge\beta)(v_{1},\ldots,v_{k+l})=\frac{1}{k!l!}\sum_{\sigma\in{}S_{k+l}}(-1)^{\sigma}\alpha(v_{\sigma(1)},\ldots,v_{\sigma(k)})\beta(v_{\sigma(k+1)},\ldots,v_{\sigma(k+l)})
	\]
}

\par\noindent
Note: We can define $(\alpha\otimes\beta)(v_{1},\ldots,v_{k+l})=\alpha(v_{1},\ldots,v_{k})\beta(v_{k+1},\ldots,v_{k+l})$, but this may not be alternating (in fact, it almost certainly isn't). But this can be skew symmetrized by forming the above sum (with appropriate normalization).\n

\par\noindent
Note: An equivalent formula for the wedge product is
\[
	(\alpha\wedge\beta)(v_{1},\ldots,v_{k+l})=\sum_{\sigma\in\Sh(k,l)}(-1)^{\sigma}\alpha(v_{\sigma(1)},\ldots,v_{\sigma(k)})\beta(v_{\sigma(k+1)},\ldots,v_{\sigma(k+l)})
\]
This is a much smaller sum, as it removes redundancies.\n

\par\noindent
Recall: $\sigma\in\Sh(k,l)$ iff $\sigma\in{}S_{k+l}$, $\sigma(1)<\cdots<\sigma(k)$, $\sigma(k+1)<\cdots<\sigma(k+l)$.\n
Note: $\#\Sh(k,l)=\binom{k+l}{k}=\binom{k+l}{l}$.\n

\ex{
	Say $\alpha\in\bigwedge^{2}V^{*}$, $\beta\in\bigwedge^{2}V^{*}$. Then the elements of $\Sh(2,2)$ are:
	\[
		\begin{array}{cc|cc|c}
			1 & 2 & 3 & 4 & \sgn\\ \hline
			1 & 2 & 3 & 4 & +\\
			1 & 3 & 2 & 4 & -\\
			1 & 4 & 2 & 3 & -\\
			2 & 3 & 1 & 4 & +\\
			2 & 4 & 1 & 3 & +\\
			3 & 4 & 1 & 2 & +
		\end{array}
	\]
	So
	\begin{align*}
		(\alpha\wedge\beta)(v_{1},v_{2},v_{3},v_{4})={} & \alpha(v_{1},v_{2})\beta(v_{3},v_{4})\\
		& -\alpha(v_{1},v_{3})\beta(v_{2},v_{4})\\
		& -\alpha(v_{1},v_{4})\beta(v_{2},v_{3})\\
		& +\alpha(v_{2},v_{3})\beta(v_{1},v_{4})\\
		& +\alpha(v_{2},v_{4})\beta(v_{1},v_{3})\\
		& +\alpha(v_{3},v_{4})\beta(v_{1},v_{2})
	\end{align*}
}

\subsubsection*{Properties of the Wedge Product}

\begin{itemize}
	\item The wedge product is bilinear: $(\alpha_{1}+\lambda\alpha_{2})\wedge\beta=\alpha_{1}\wedge\beta+\lambda(\alpha_{2}\wedge\beta)$.
	\item The wedge product is associative: $\alpha\wedge(\beta\wedge\gamma)=(\alpha\wedge\beta)\wedge\gamma=\alpha\wedge\beta\wedge\gamma$.
	\item The wedge product is anticommutative: for $\alpha\in\bigwedge^{k},\beta\in\bigwedge^{l}$, $\alpha\wedge\beta=(-1)^{kl}\beta\wedge\alpha$.\n
	This implies that even forms commute with any other form.
	\item If $\alpha^{1},\ldots,\alpha^{k}\in{}V^{*}$, then $(\alpha^{1}\wedge\cdots\wedge\alpha^{k})(v_{1},\ldots,v_{k})=\det\begin{pmatrix}\alpha^{i}(v_{j})\end{pmatrix}_{(i,j)}$.\n
	In particular, $\mathcal{E}^{I}=\mathcal{E}^{i_{1}}\wedge\cdots\wedge\mathcal{E}^{i_{k}}$.
\end{itemize}

\ex{
	Say $\alpha^{1},\alpha^{2}\in{}V^{*}$. Then $(\alpha^{1}\wedge\alpha^{2})(v_{1},v_{2})=\alpha^{1}(v_{1})\alpha^{2}(v_{2})-\alpha^{1}(v_{2})\alpha^{2}(v_{1})$.\n
}

\defn{
	$\displaystyle\paren{\bigoplus_{k=0}^{n}\bigwedge^{k}V^{*},\wedge}$ is the \u{exterior algebra} or \u{Grassmann algebra} of $V$.\n
}

\par\noindent
Now, back to manifolds...\n

\defn{
	Let $M$ be a smooth manifold. A \u{$k$-differential form} (or \u{$k$-form}) on $M$ is an assignment $\forall{}p\in{}M$, $p\mapsto\alpha_{p}\in\bigwedge^{k}(T_{p}^{*}M)$.
}

\par\noindent
Note: When $k=1$, $\alpha$ is a $1$-form as before. When $k=0$, $\alpha$ is just an $\R$-valued function.\n

\subsection*{Local Expression}

\par\noindent
Let $(U,(x^{1},\ldots,x^{n}))$ be a coordinate chart. Then $\forall{}p\in{}U$, we get $\set{\restr{dx^{i}}{p}}_{i=1,\ldots,n}$ a basis of $T_{p}^{*}M$, with corresponding dual basis $\set{\pdat{}{x^{i}}{p}}$ of $T_{p}M$. If $\alpha$ is a $k$-form, then $\forall{}p\in{}U$,
\[
	\alpha_{p}={\sum_{I}}'a_{I}(dx^{I})_{p}\qquad{}a_{I}(p)=\alpha_{p}\paren{\pdat{}{x^{i_{1}}}{p},\ldots,\pdat{}{x^{i_{k}}}{p}}
\]
This defines functions $a_{I}:U\to\R$ for each $I=\set{i_{1}<\cdots<i_{k}}\subseteq\set{1,\ldots,n}$.\n

\defn{
	A $k$-form $\alpha$ is \u{smooth} iff there exists a $C^{\infty}$ atlas of $M$ such that in each chart, each $a_{I}$ is smooth.\n
}

\newpage
\lemma{
	$\alpha$ is smooth iff for every chart in the smooth structure, each function $a_{I}$ is smooth.\nn
	Proof: $\Leftarrow$ is trivial. For $\Rightarrow$, let $(y^{1},\ldots,y^{n}):V\to\R$ be an arbitrary coordinate chart. Let $p\in{}V$. By our assumption, there's a chart $(x^{1},\ldots,x^{n}):U\to\R$ near $p$ such that $\alpha={\sum_{I}}'a_{I}dx^{I}$, with $a_{I}\in{}C^{\infty}(U)$. Also, $\alpha={\sum_{J}}'b_{J}dy^{J}$. We need to show $b_{J}\in{}C^{\infty}(U\cap{}V)$, but how? Well,
	\[
		b_{J}=\alpha\paren{\pd{}{y^{j_{1}}},\ldots,\pd{}{y^{j_{k}}}}={\sum_{I}}'a_{I}\underbrace{dx^{I}\paren{\pd{}{y^{j_{1}}},\ldots,\pd{}{y^{j_{k}}}}}_{\hspace{-9em}\mathrlap{\substack{
			\displaystyle=\det\begin{pmatrix}dx^{i_{r}}\paren{\pd{}{y^{j_{s}}}}\end{pmatrix}_{(r,s)}\\ \\
			\displaystyle=\det\begin{pmatrix}\pd{x^{i_{r}}}{y^{j_{s}}}\end{pmatrix}_{(r,s)}
		}}}
	\]
	The $\pd{x^{i_{r}}}{y^{j_{s}}}$ are smooth, so the whole determinant is smooth, so $b_{J}$ is the sum of smooth functions.\n
	Thus, it's smooth.\proven
