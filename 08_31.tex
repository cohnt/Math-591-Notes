\par\noindent
What \underline{is} a differentiable manifold? It's a bit of a long answer. Here's something that's true, but not complete:\n

\par\noindent
A differentiable manifold is a topological space that is
\begin{enumerate}[label=(\alph*), topsep=0pt, itemsep=0pt]
	\item Hausdorff ($T_{2}$)
	\item Second-Countable
	\item Locally Euclidean (Informally, this means every point has a neighborhood that is homeomorphic to a Euclidean space.)
	\item Has a $C^{\infty}$ atlas (This is additional structure, not a topological property.)
\end{enumerate}
Note that (a), (b), and (c) together define a topological manifold.\n

\subsection*{Hausdorff Spaces}

\defn{
	A topological space $X$ is \underline{Hausdorff} (or \underline{$T_{2}$}) if $\forall{}x,y\in{}X$ with $x\ne{}y$, there exist neighborhoods $U$ of $x$ and $V$ of $y$ such that $U\cap{}V=\emptyset$.\n
}

\ex{
	Any metric space is Hausdorff.\n
	Proof: Fix distinct $x,y$. Let $r=\frac{1}{3}d(x,y)>0$ (because $x\ne{}y$). Let $U=B(x,r)$, $V=B(y,r)$. Then by the triangle inequality, $U\cap{}V=\emptyset$.\proven
}

\ex{
	The following topological space is \textit{not} Hausdorff.\n
	Let $X=(-\infty,0)\cup\set{A,B}\cup(0,\infty)$ (with $A\ne{}B$), with the standard topology on $(-\infty,0)$ and $(0,\infty)$, and neighborhoods of $A$ and $B$ contain $(-\varepsilon,0)$ and $(0,\varepsilon)$.\n
	This space is not Hausdorff, because you can't separate $A$ and $B$.\n
}

\subsection*{Second-Countable Spaces}

\defn{
	Let $X$ be a topological space. A \underline{basis} of $X$ is a collection $\mathscr{B}$ of open sets such that every open set in $X$ can be written as the (possibly infinite) union of elements in $\mathscr{B}$.\n
}

\defn{
	$X$ is \underline{second-countable} if there is a countable basis of $X$.\n
}

\ex{
	$X=\R^{n}$ (with the standard topology) is second countable.\n
	Proof: Let $\mathscr{B}=\set{B(q,r)\mid{}q\in\Q^{n},r\in\Q}$. We know $\mathscr{B}$ is countable, so we need to show $\mathscr{B}$ is a basis.\n
	Let $U\subseteq\R^{n}$ be open. Let $\mathscr{B}_{U}=\set{B\in\mathscr{B}\mid{}B\subseteq{}U}$. We claim that $U=\bigcup_{B\in\mathscr{B}_{U}}B$.\n
	Well, $\supseteq$ is trivial. To show $\subseteq$, let $x\in{}U$. Since $U$ is open, $\exists{}r\in\R_{>0}$ \st{} $B(x,r)\subseteq{}U$. Let $q\in\Q^{n}$ \st{} $d(x,q)<\frac{r}{43}$. Let $\rho\in\Q$ such that $\frac{r}{43}\overset{\ptxt{(1)}}{<}\rho\overset{\ptxt{(2)}}{<}\frac{r}{42}$.
	\begin{enumerate}[label=(\arabic*), itemsep=0pt, leftmargin=4\parindent]
		\item implies that $B(q,\rho)$ contains $x$.
		\item implies (by the triangle inequality and the fact that $x\in{}B(q,\rho)$) that $B(q,\rho)\subseteq{}U$, so $B(q,\rho)\in\mathscr{B}_{U}$.
	\end{enumerate}\up\n
	\proven
}

\ex{
	The following topological space is \textit{not} second-countable.\n
	Let $X=\R$ with the discrete topology (i.e. singletons are open).\n
	A basis must have every singleton, so it's obviously not second-countable.\n
}

\subsubsection*{Product Spaces}

\defn{
	Let $X$, $Y$ be topological spaces. Then $X\times{}Y$ has a natural topology called the \underline{product topology}, defined by the basis $\set{U\times{}V\mid{}U\subseteq{}X,V\subseteq{}Y\ptxt{ open}}$.\n
}

\thm{
	(A.21, A.22) If $X$ and $Y$ are second-countable topological spaces, then so is $X\times{}Y$. If $X$ and $Y$ are $T_{2}$, then so is $X\times{}Y$.\n
}
