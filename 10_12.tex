\thm{
	(Normal Form for Immersions) Let $F:M\to{}N$ be an immersion at $p\in{}M$. Then there are coordinates $p\in{}U,\phi=(x^{1},\ldots,x^{m})$ of $M$ and $f(p)\in{}V,\psi=(y^{1},\ldots,y^{n})$ of $N$ such that $U\subset{}F\inv(V)$ and $\tilde{F}(r^{1},\ldots,r^{m})=(r^{1},\ldots,r^{n},\underbrace{0,\ldots,0}_{n-m\ptxt{ zeros}})$.\nn
	Proof: take any coordinates. We have
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			M\supseteq{}U \arrow[r, "F"] \arrow[d, "\phi"] \& V\subseteq{}N \arrow[d,"\psi"]\\
			\R^{m}\supseteq\phi(U) \arrow[r,"\tilde{F}"] \& \psi(V)\subseteq{}\R^{n}
		\end{tikzcd}
	\]
	Write $F=(F^{1},\ldots,F^{n})$, where each $F^{i}=y^{i}\of{}F$.\n
	The Jacobian of $\tilde{F}$ at $\phi(p)$ is $J=\begin{pmatrix}\pd{F^{i}}{x^{j}}(p)\end{pmatrix}_{\substack{i\ptxt{ rows}\\ j\ptxt{ cols}}}$. By assumption, $\ker{}J=\set{0}$, so we can write
	\[
		J=\begin{pmatrix}
			\mathcal{M}_{m\times{}m}\\
			\star
		\end{pmatrix}_{\substack{m\le{}n\\ 1\le{}i\le{}n\\ 1\le{}j\le{}m}}
	\]
	and we know $J$ has max rank. Permute the $y$-coordinates to shuffle the rows, so that the top $m\times{}m$ minor $\mathcal{M}$ of $J$ is non-degenerate. Thus, $\mathcal{M}=\begin{pmatrix}\pd{F^{i}}{x^{j}}\end{pmatrix}_{\substack{1\le{}i\le{}m\\ 1\le{}j\le{}m}}$ gives us some $\tilde{\phi}=(F^{1},\ldots,F^{m}):U\to\R^{m}$ which is a local diffeomorphism at $p$. By shrinking $U$ if necessary, $\tilde{\phi}:U\to\R^{m}$ is a coordinate chart.
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			M\supseteq{}U \arrow[r,"F"] \arrow[d,"\tilde\phi"] \& V\subseteq{}N \arrow[d,"\psi"]\\
			\tilde\phi(U) \arrow[r,"\tilde{\tilde{F}}"]\& \psi(V)
		\end{tikzcd}
	\]
	where $\tilde{\tilde{F}}(\underbrace{r^{1},\ldots,r^{m}}_{r})=(y^{1}(F\of\tilde\phi(r)),\ldots,y^{n}(F\of\tilde\phi(r)))=(r^{1},\ldots,r^{m},G^{1}(r),\ldots,G^{n-m}(r))$. Now, we modify the $y$-coordinates:
	\[
		\left\{\begin{array}{ll}
			w^{i}=y^{i} & 1\le{}i\le{}m\\
			w^{i}=y^{i}-G^{i-m}(y^{1},\ldots,y^{m}) & m+1\le{}i\le{}n
		\end{array}\right.
	\]
	Both are invertible. The first $m$ are trivially so, and the remainder are invertible because you can recover $y^{i+m}$.\n
	Let $\tilde\psi=(w^{1},\ldots,w^{n}):V\to\R^{n}$.
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			\&[-20pt] V \arrow[dl, "\psi", swap] \arrow[dr, "\phi"] \&[-20pt]\\[-5pt]
			\R^{n} \arrow[rr, "\star"] \& \& \R^{n}
		\end{tikzcd}
	\]
	with
	\[
		\star:(r^{1},\ldots,r^{n})\mapsto(\underbrace{r^{1},\ldots,r^{m}}_{r^{'}},\underbrace{r^{m+1}-G^{1}(r'),\ldots,r^{n}-G^{m-n}(r')}_{r''-G(r')})
	\]
	This is a local diffeomorphism. So under the $\tilde\phi$ and $\tilde\psi$ coordinates, $(r^{1},\ldots,r^{m})\overset{\tilde{\tilde{\tilde{F}}}}{\to}(r^{1},\ldots,r^{m},0,\ldots,0)$.\proven
}

\par\noindent
Recall: An embedding $F:M\to{}N$ is an immersion that is a homeomorphism onto its image, that is, $F|^{F(M)}:M\to{}F(M)$ is a homeomorphism. It has a continuous inverse.\n

\par\noindent
Observe: Every embedding is also one-to-one.\n

\par\noindent
Question: Assum $F$ is a one-to-one immersion. Under what conditions is $F$ an embedding?\n

\prop{
	An injective immersion is an embedding iff $F|^{F(M)}:M\to{}F(M)$ is an open (or closed) map w.r.t. the subspace topology. This is true iff $\forall{}U\subset{}M$ open, $\exists{}V\subset{}N$ open \st{} $F(U)=F(M)\cap{}V$ ($F(M)\cap{}V)$ is a relatively open set).\nn
	Proof: The inverse of an open map is continuous iff the map is invertible.\proven
}

\cor{
	If $F:M\to{}N$ is an injective immersion and $M$ is compact, then $F$ is an embedding.\nn
	Proof: $F$ is a closed map -- if $C\subset{}M$ is closed and $M$ is compact, then $C$ is compact, so $F(C)$ is compact. Thus, $F(C)$ is closed.\proven
}

\par\noindent
In fact...\n

\thm{
	If $F:M\to{}N$ is a proper (recall: the preimage of compact sets are compact) injective immersion, then $F$ is an embedding.\nn
	``Proof'': This is true because a proper continuous map into a locally compact space is closed.\n
}

\thm{
	The image of an embedding is a regular submanifold.\nn
	Proof: Let $F:M\to{}N$ be an embedding. Let $q=F(p)\in{}F(M)$ for $p\in{}M$. Use the immersion theorem to get coordinate $(x^{1},\ldots,x^{m})$ and $(y^{1},\ldots,y^{n})$ \st{} $\tilde{F}(r^{1},\ldots,r^{m})=(r^{1},\ldots,r^{n},\underbrace{0,\ldots,0}_{n-m})$.\n
	Since $F$ is an embedding, $F(U)$ is relatively open in $F(M)$. Thus, $F(U)$ is the intersection of $F(M)$ with some open set $\tilde{V}$ in $N$. Thus, $F(U)=\tilde{V}\cap{}F(M)\subseteq\set{y^{m+1}=\cdots=y^{n}=0}$. (Will finish next time.)
}
