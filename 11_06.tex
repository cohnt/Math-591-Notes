\par\noindent
Recall: For $G$ a Lie group, $\mathfrak{g}=T_{e}G$ is its Lie algebra. $\forall{}A\in\mathfrak{g}$, $A^{\sharp}$ is the left-invariant vector field on $G$ determined by $A$, that is, $\forall{}g\in{}G$, $A^{\sharp}_{g}\eqdef(L_{g})_{*,e}(A)$. In particular, $A^{\sharp}_{e}=A$.\n

\defn{
	$\forall{}A\in\mathfrak{g}$, $t\in\R$, $\exp{}t\,A$ is the integral curve of $A^{\sharp}$ at $e$, at time $t$.\n
}

\defn{
	$\map{\exp:\mathfrak{g}}{G}{A}{\restr{\exp{}t\,A}{t=1}}$\n
}

\prop{
	$\exp$ is smooth.\nn
	Proof: Introduce the field $X\in\mathfrak{X}(G\times\mathfrak{g})$, defined by, $\forall{}(g,A)\in{}G\times\mathfrak{g}$,
	\[
		X_{(g,A)}=(A^{\sharp}_{g},0)\in{}T_{g}G\times{}T_{A}\mathfrak{g}=T_{(g,A)}(G\times\mathfrak{g})=T_{g}G\times{}\mathfrak{g}
	\]
	$X$ is smooth. Its flow, denoted $F$, is $F_{t}(g,A)=(F_{t}^{A}(g),A)$, where $F^{A}$ is the flow of $A^{\sharp}$. Thus, the time-$1$ map is $C^{\infty}$.

\prop{
	With the same notation -- $A\in{}\mathfrak{g}$, $F^{A}$ the flow of $A^{\sharp}$ -- $\forall{}g\in{}G$, $F_{t}^{A}(g)=g\cdot\exp{}t\,A$ (with $\cdot$ group multiplication).\nn
	Proof: $\frac{d}{dt}(g\cdot\exp{}t\,A)=\frac{d}{dt}L_{g}(\exp{}t\,A)=(L_{g})_{*,\exp(A)}(A^{\sharp}_{\exp{}t\,A})=A^{\sharp}_{g\cdot\exp{}t\,A}$.\n
	This is precisely the ODE satisfied by $t\mapsto{}F^{A}_{t}(g)$.\proven
}

\defn{
	If $H,G$ are Lie groups, then a \u{Lie group morphism} from $H$ to $G$ is a smooth map $F:H\to{}G$ that is also a group homomorphism.\n
}

\prop{
	Let $F:H\to{}G$ be a Lie group homomorphism. Then
	\begin{enumerate}[label=\alph*), leftmargin=4\parindent]
		\item $\forall{}A\in\mathfrak{h}=T_{e}H$, $A^{\sharp}\in\mathfrak{X}(H)$ and $[\underbrace{F_{*,e}(A)}_{\in{}T_{e}G=\mathfrak{g}}]^{\sharp}\in\mathfrak{X}(G)$. These two fields are $F$-related.
		\item $\forall{}A\in\mathfrak{h}$, $F(\underbrace{\exp{}}_{\ptxt{in }H}A)=\underbrace{\exp}_{\ptxt{in }G}(F_{*,e}(A))$.
		\item $F_{*,e}:\mathfrak{h}\to\mathfrak{g}$ is a Lie algebra morphism. That is, $\forall{}A,B\in\mathfrak{h}$, $F_{*,e}(\brack{A,B})=\brack{F_{*,e}(A),F_{*,e}(B)}$ (with the appropriate Lie brackets).
	\end{enumerate}\up\n
	Proof: (b) and (c) follow directly from (a).\n
	Claim: $\forall{}h\in{}H$, the following diagram commutes:
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			H \arrow[r, "F"] \arrow[d, "L_{h}", swap] \& G \arrow[d, "L_{F(h)}"]\\
			H \arrow[r, "F", swap] \& G
		\end{tikzcd}
	\]
	Check: let $k\in{}H$. Then
	\[
		(F\of{}L_{h})(k)=F(hk)=F(h)F(k)=L_{F(h)}(F(k))
	\]
	because $F$ is a group morphism. Thus, $\forall{}h$, $F_{*,h}\of(L_{h})_{*,e}=(L_{F(h)})_{*,F(h)}\of{}F_{*,e}$ by the chain rule, and commutativity at the identity). We can apply this to any $A\in\mathfrak{h}$:
	\[
		F_{*,h}(\underbrace{(L_{h})_{*,e}(A)}_{=A^{\sharp}_{h}})=(L_{F(h)})_{*,F(e)}(F_{*,e}(A))=\underbrace{(L_{F(h)})_{*,e}(F_{*,e}(A))}_{=[F_{*,e}(A)]^{\sharp_{F(h)}}}
	\]
	This \i{is} the fact that $A^{\sharp}$ and $[F_{*,e}(A)]^{\sharp}$ are $F$-related.\proven
}

\par\noindent
Now, back to subgroups.\n

\defn{
	A \u{regular} (or \u{embedded}) \u{Lie subgroup} of $G$ is a regular submanifold that is also a subgroup.\n
}

\cor{
	If $H\subseteq{}G$ is a regular subgroup, then
	\begin{enumerate}[leftmargin=4\parindent, label=(\alph*)]
		\item $\mathfrak{h}\subset\mathfrak{g}$ as a subspace is closed under $\brack{\,\cdot\,,\,\cdot\,}$ of $\mathfrak{g}$.
		\item $\forall{}A\in\mathfrak{h}$, $\underbrace{\exp}_{\ptxt{in }H}A=\underbrace{\exp}_{\ptxt{in }G}A$.
	\end{enumerate}\up\n
	Proof: Just apply the previous theorem to $F:H\hookrightarrow{}G$.\proven
}

\par\noindent
Observe: $\forall{}A,B\in\mathfrak{h}$, $\brack{A,B}_{\mathfrak{h}}=\brack{A,B}_{\mathfrak{g}}$, so $\brack{\,\cdot\,,\,\cdot\,}_{\mathfrak{h}}=\restr{\brack{\,\cdot\,,\,\cdot\,}_{\mathfrak{g}}}{\mathfrak{h}}$.\n

\defn{
	Given a Lie algebra $\mathfrak{g}$, a \u{Lie-subalgebra} is a subspace $\mathfrak{h}\subset\mathfrak{g}$ which is closed under $\brack{\,\cdot\,,\,\cdot\,}$.\n
}

\cor{
	If $H\subset{}G$ is a regular Lie subgroup, then $\mathfrak{h}=T_{e}H$ is a Lie-subalgebra of $\mathfrak{g}$.\n
}

\par\noindent
Is the converse true? I.e., given $G$ a Lie group, if $\mathfrak{h}\subset{}T_{e}G=\mathfrak{g}$ is a Lie subalgebra of $\mathfrak{g}$, is there a Lie subgroup $H\subset{}G$ such that $T_{e}H=\mathfrak{h}$?\n
Answer: Yes, if we allow immersed submanifolds.\n

\thm{
	If $\mathfrak{h}\subset{}T_{e}G$ is a Lie subalgebra, then there is a Lie group $H$ and an immersion $F:H\to{}G$ which is a group morphism and $F_{*,e}:T_{e}H\overset{\cong}{\to}\mathfrak{h}$.\n
}

\ex{
	An irrational line on the two-torus $S^{1}\times{}S^{1}=\mathbb{T}^{2}$,
	\[
		\map{F:\R}{\mathbb{T}^{2}}{t}{(e^{it},e^{i\alpha{}t})}
	\]
	for $\alpha$ irrational, is a Lie group morphism and an immersion. $\im(F_{*,e})$ is a $1$-dimensional subspace of $\mathfrak{t}^{2}\cong\R^{2}$. This is always a subalgebra.\n
}
