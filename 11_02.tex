\section*{Lie Derivatives}

\par\noindent
The general notion of a Lie derivative with respect to a vector field $X$ is ``pull back by $\theta_{t}$, the time $t$ map of the flow of $X$, and then differentiate with respect to $t$ at $t=0$''. We call this $\mathcal{L}_{x}$.\n

\ex{
	We can pull back functions, so we can have Lie derivatives of functions.\n
	Given $X$, $\theta_{t}$, and $f\in{}C^{\infty}(M)$, $\phi_{t}^{*}(f)(p)=f(\phi_{t}(p))$ (pullback). Now differentiate:
	\[
		\left.\frac{d}{dt}\phi_{t}^{*}(f)(p)\right|_{t=0}=\left.\frac{d}{dt}f(\phi_{t}(p))\right|_{t=0}=X(f)(p)
	\]
	So $(\mathcal{L}_{X}f)(p)=X(f)(p)=df(X_{p})$. So $\mathcal{L}_{X}f$ is again a function on $M$.\n
}

\par\noindent
Recall: A $1$-form $\alpha$ is an assignment to each $p\in{}M$, an element $\alpha_{p}\in{}T_{p}^{*}M$. I.e. $\alpha$ is a section of the cotangent bundle. In local coordinates, we can write $\alpha=\sum{}\alpha_{i}dx^{i}$, for $\alpha_{i}\in{}C^{\infty}(U)$.\n

\ex{
	$1$-forms can be pulled back by any $F$, by $F^{*}(\alpha)(v)=\alpha(F_{*,\cdot}(v))$. So
	\[
		(\mathcal{L}_{X}\alpha)(p)=\frac{d}{dt}\underbrace{\phi_{t}^{*}\alpha_{\phi_{t}(p)}}_{\substack{\in{}T_{p}^{*}M\\ \ptxt{ for each $t$}}}\left.\vphantom{\frac{d}{dt}}\right|_{t=0}
	\]
	Thus, $\mathcal{L}_{X}\alpha$ is again a $1$-form on $M$.\n
}

\par\noindent
Special case: $\alpha=df$ for some $f\in{}C^{\infty}(M)$. Then by the chain rule (1)
\[
	F^{*}(df)(v)=df(F_{*,\cdot}(v))\overset{\mathclap{\ptxt{(1)}}}{=}d(f\of{}F)(v)=d(F^{*}f)(v)
\]
so $F^{*}(df)=d(F^{*}f)$.\n

\par\noindent
So, in general,
\[
	\mathcal{L}_{X}(df)=\left.\frac{d}{dt}\phi_{t}^{*}(df)\right|_{t=0}=\restr{\frac{d}{dt}d\phi_{t}^{*}(f)}{t=0}\labeledeq{(1)}d\paren{\restr{\frac{d}{dt}\phi_{t}^{*}f}{t=0}\,}=d(df(X))=d(X(f))
\]
Where (1) is true because $\frac{d^{2}}{\d{}t\d{}x^{i}}=\frac{\d^{2}}{\d{}^{i}\d{}t}$, so the differentiations commute.\n
In fact, in general, $\mathcal{L}_{X}(\alpha)=d(\alpha(X))+\cdots$ (we'll fill in the other term later).\n

\par\noindent
Now, back to vector fields...\n

\par\noindent
Note: Vector fields can be pulled back by diffeomorphisms. This means if $X,Y\in\mathfrak{X}(M)$, $\mathcal{L}_{X}Y$ is defined by, $\forall{}p\in{}M$,
\[
	\mathcal{L}_{X}(Y)(p)=\underbrace{\frac{d}{dt}\brack{(\phi_{t})_{*,p}}\inv(Y_{\phi_{t}(p)})}_{=V_{t}\in{}T_{p}M}\left.\vphantom{\frac{d}{dt}}\right|_{t=0}
\]

\newpage
\prop{
	$\mathcal{L}_{X}Y=\brack{X,Y}$.\nn
	Proof: Compute $V_{t}$ in coordinates $(U,(x^{1},\ldots,x^{n}))$. Restrict to $V\subseteq{}U$ such that $\phi_{t}(V)\subseteq{}U$. Write $\phi_{t}=(\phi_{t}^{1},\ldots,\phi_{t}^{n})$. Then
	\[
		J_{t}=\begin{pmatrix}
			\pd{\phi_{t}^{i}}{x^{j}}(p)
		\end{pmatrix}_{(i,j)}
		\quad
		Y=\sum_{i=1}^{n}g_{i}\pd{}{x^{i}}
		\quad
		X=\sum_{i=1}^{n}f_{i}\pd{}{x^{i}}
	\]
	For a fixed $p$, let
	\[
		\tilde{Y}_{t}=Y_{\phi_{t}(p)}=\begin{pmatrix}
			g_{1}(\phi_{t}(p))\\
			\vdots\\
			g_{n}(\phi_{t}(p))
		\end{pmatrix}
	\]
	Then, let $\tilde{V}_{t}$ be the column vector with components of $V_{t}$. We have
	\[
		\tilde{V}_{t}=J_{t}\inv\tilde{Y}_{t}
	\]
	Differentiating with $\frac{d}{dt}$ on both sides, we get
	\[
		\dot{\tilde{V}}_{t}=J_{t}\inv\dot{\tilde{Y}}_{t}-J_{t}\inv\dot{J}_{t}J_{t}\inv\tilde{Y}_{t}
	\]
	which at $t=0$, yields
	\[
		\underbrace{\left.\dot{\tilde{V}}_{t}\right|_{t=0}}_{\substack{\ptxt{components}\\ \ptxt{of $\mathcal{L}_{X}Y$}}}=\begin{pmatrix}
			X(g_{1})(p)\\
			\vdots\\
			X(g_{n})(p)
		\end{pmatrix}-?
	\]
	Well,
	\[
		\dot{J}_{t}=\begin{pmatrix}
			\frac{\d^{2}\phi_{t}^{i}}{\d{}t\d{}x^{j}}(p)
		\end{pmatrix}_{(i,j)}=\begin{pmatrix}
			\frac{\d^{2}\phi_{t}^{i}}{\d{}x^{j}\d{}t}(p)_{(i,j)}
		\end{pmatrix}=\begin{pmatrix}
				\pd{f_{i}}{x^{j}}(p)
			\end{pmatrix}_{(i,j)}
	\]
	since $\pd{\phi_{t}^{i}}{t}(p)=f_{i}$. Thus, we have
	\[
		\left.\dot{\tilde{V}}_{t}\right|_{t=0}=\begin{pmatrix}
			X(g_{1})(p)\\
			\vdots\\
			X(g_{n})(p)
		\end{pmatrix}-\begin{pmatrix}
			\pd{f_{i}}{x^{j}}(p)
		\end{pmatrix}_{(i,j)}\begin{pmatrix}
			g_{1}(p)\\
			\vdots\\
			g_{n}(p)
		\end{pmatrix}
	\]
	Therefore,
	\begin{align*}
		\mathcal{L}_{X}Y & =\sum_{i=1}^{n}X(g_{i})\pd{}{x^{i}}-\sum_{i,j=1}^{n}\pd{f_{i}}{x^{j}}g_{j}\pd{}{x^{i}}\\
		& =\sum_{i=1}^{n}X(g_{i})\pd{}{x^{i}}-\sum_{i=1}^{n}Y(f_{i})\pd{}{x^{i}}\\
		& =\sum_{i=1}^{n}(X(g_{i})-Y(f_{i}))\pd{}{x^{i}}\\
		& =\brack{X,Y}
	\end{align*}
	\proven
}

\thm{
	Using the same notation as last time, $\forall{}t\in(-\varepsilon,\varepsilon)$, $V_{t}=\frac{d}{dt}(\phi_{t,*})\inv(Y_{\phi_{t}}(p))=(\phi_{t,*})\inv[X,Y]_{\phi_{t}(p)}$.\nn
	Proof: We already have it for $t=0$. Now, consider $V_{t+s}$, and use the group law/translation lemma.\proven
}

\cor{
	If $\brack{X,Y}=0$, then the flows of $X$ and $Y$ commute.\n
}
