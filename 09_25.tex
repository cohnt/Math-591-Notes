\subsection*{Tangent Vectors}

\par\noindent
Last time, we proved that for $p\in{}U\osubseteq{}M$, $\phi:U\to\R^{n}$ chart, $\phi=(x^{1},\ldots,x^{n})$, that $\forall{}v\in{}T_{p}M$, we can write
\[
	v=\sum_{i=1}^{n}v([x^{i}])\pdat{}{x^{i}}{p}
\]
This is based on:

\thm{
	If $g:B\to\R$, with $B\subseteq\R^{n}$ being the open ball centered at the origin, then there exist $g_{ij}\in{}C^{\infty}(B)$ \st{} $\forall{}r\in{}B$,
	\[
		g(r)=g(0)+\sum_{j=1}^{n}r^{j}\pd{g}{r^{j}}(0)+\frac{1}{2}\sum_{i,j=1}^{n}r^{i}r^{j}g_{ij}(r)
	\]
	with $g_{ij}(0)=\pd{^{2}g}{r^{i}\partial{}r^{j}}(0)$.\nn
	Proof: Start with $g(r)=g(0)+\int_{0}^{1}\frac{d}{dt}g(tr)\,dt$. Then by the fundamental theorem of calculus, this is equal to
	\[
		=g(0)+\int_{0}^{1}\sum_{j=1}^{n}r^{j}\pd{g}{r^{j}}(tr)\,dt=g(0)+\sum_{j=1}^{n}r^{j}\hspace{-4pt}\underbrace{\int_{0}^{1}\pd{g}{r^{j}}(tr)\,dt}_{g_{j}(r)\in{}C^{\infty}}=g(0)+\sum_{j=1}^{n}r^{j}g_{j}(r)
	\]
	We can then repeat this argument with each $g_{j}$, so for each $j$, there are some $g_{ji}\in{}C^{\infty}$ \st{}
	\[
		g_{j}(r)=g_{j}(0)+\sum_{i=1}^{n}r^{i}g_{ji}(r)
	\]
	(The exact computation may be off here by a factor of $2$, due to symmetry.)\n
	Observe that
	\[
		g_{j}(0)=\int_{0}^{1}\pd{g}{r^{j}}(0)\,dt=\pd{g}{r^{j}}(0)
	\]
	Plugging the $g_{j}$'s back in, we get
	\[
		g(r)=g(0)+\sum_{j=1}^{n}r^{j}g_{j}(0)+\sum_{j=1}^{n}r^{j}\sum_{i=1}^{n}r^{i}g_{ji}(r)=g(0)+\sum_{j=1}^{n}r^{j}g_{j}(0)+\sum_{i,j=1}^{n}r^{j}r^{i}g_{ji}(r)
	\]
	\proven
}

\subsection*{Tangent Vectors and Curves}

\par\noindent
Let $p\in{}U\osubseteq{}M$, $\phi:U\to\R^{n}$ chart, $\phi=(x^{1},\ldots,x^{n})$. Then let $\gamma$ so that
\[
	\begin{tikzcd}
		(-\varepsilon,\varepsilon) \arrow[r, "\gamma"] \arrow[dr, swap, "\phi\of\gamma"] & U \arrow[d, "\phi"] \arrow[dr, "f"]\\
		& \R^{n} \arrow[r, swap, "f_{\phi}"] & \R
	\end{tikzcd}
\]
Previously, we defined $\dot\gamma(0)\in{}T_{p}M$ so that $\dot\gamma(0)([f])=\restr{\frac{d}{dt}(f\of\gamma)}{t=0}$, where $f\of\gamma:(-\varepsilon,\varepsilon)\to\R$.\n

\par\noindent
Computation of $\dot\gamma(0)$ in coordinates:\n

\lemma{
	Let $(\phi\of\gamma)(t)=(x^{1}(t),\ldots,x^{n}(t))$, defined by $x^{i}(t):(-\varepsilon,\varepsilon)\to\R$. Then
	\[
		\dot\gamma(0)=\sum_{j=1}^{n}\restr{\frac{dx^{j}(t)}{dt}}{t=0}\pdat{}{x^{j}}{p}
	\]
	Proof: Let $f:U\to\R$. Then $f\of\gamma=(f\of\phi\inv)\of(\phi\of\gamma)=f_{\phi}\of(\phi\of\gamma)$. Use the chain rule on the right-hand side. Then
	\[
		\dot\gamma(0)[f]=\left.\frac{d}{dt}(f\of\gamma)(t)\right|_{t=0}=\sum_{j=1}^{n}\underbrace{\pd{f_{\phi}}{r^{j}}((\phi\of\gamma)(t))}_{\pd{f}{x^{j}}(\gamma(t))}\left.\frac{dx^{j}(t)}{dt}\right|_{t=0}
	\]
	\proven
}

\cor{
	Any $v\in{}T_{p}M$ is equal to $\dot\gamma(0)$ for some curve $\gamma$.\n
	Proof: Choose a chart $(U,\phi)$ so that $\phi(p)=0$. Then $v=\sum_{j=1}^{n}v_{j}\pdat{}{x^{j}}{p}$, with each $v_{j}\in\R$. Define $\gamma$ by $x^{j}(t)=tv_{j}$, $\forall{}j\in\set{1,\ldots,n}$, and letting this define $\phi\of\gamma$.
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			(-\varepsilon,\varepsilon) \arrow[r,"\gamma"] \arrow[dr,swap,"\phi\of\gamma"] \& U \arrow[d,"\phi"]\\
			\& \R^{n}
		\end{tikzcd}
	\]
	Then $\gamma(p)=\phi\inv(tv_{1},\ldots,tv_{n})$.\proven
}

\subsection*{Smooth Maps Between Manifolds and Tangent Spaces}

\par\noindent
Let $F:M\to{}N$ be a smooth map between smooth manifolds $M$ and $N$. Let $p\in{}M$, with $q=F(p)\in{}N$.\n

\par\noindent
Observe: Given any $f:V\to\R$, $q\in{}V\osubseteq{}N$, we have
\[
	\begin{tikzcd}
		M \arrow[r,"F"] & V \arrow[r,"f"] & \R
	\end{tikzcd}
\]
\defn{
	Consider $F\inv(V)$, an open neighborhood of $p$. $f\of{}F:F\inv(V)\to\R$. This gives us a map
	\[
		\map{F^{*}:C^{\infty}_{q}(N)}{C^{\infty}_{p}(M)}{{}[f]}{[f\of{}F]}
	\]
	This is the \u{pullback map on germs}. Note that this is a ring morphism!\n
}

\par\noindent
By duality, we can pushforward tangent vectors.\n

\defn{
	If $v\in{}T_{p}M$, we define the \u{pushforward} of $v$, $F_{*,p}(v):C_{q}^{\infty}(N)\to\R$ by $F_{*,p}(v)([f])=v(F^{*}([f]))\in\R$.\n
}

\par\noindent
Claim: $F_{*,p}(v)\in{}T_{q}N$, i.e., $F_{*,p}(v)$ is also a derivation.\n
Rough proof: Recall that $F^{*}$ is a ring morhpism. This, combined with the fact that $v$ is a derivation, implies that $F_{*,p}(v)$ is a derivation.\proven

\par\noindent
Conclusion: We obtain $F_{*,p}:T_{p}M\to{}T_{F(p)}N$.\n

\defn{
	We can take its dual: $F_{p}^{*}:T_{F(p)}^{*}N\to{}T_{p}^{*}M$.\n
}

\lemma{
	$F_{p}^{*}$ is linear.\n
}

\lemma{
	$F_{*,p}(\dot\gamma(0))=\left.\frac{d}{dt}(F\of\gamma)\right|_{t=0}$.\n
}

\par\noindent
This final lemma is very useful for computation!
