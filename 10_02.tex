\defn{
	Let $F:M\to{}N$ be smooth.
	\begin{itemize}
		\item $F$ is a \u{submersion at $p\in{}M$} $\Leftrightarrow$ $F_{*,p}$ is onto. We say $p$ is a \u{regular point} of $F$.
		\item $F$ is a submersion $\Leftrightarrow$ $\forall{}p\in{}M$, $F$ is a submersion at $p$.
		\item A \u{critical point} is a point which is not a regular point.
		\item $q\in{}N$ is a \u{regular value} $\Leftrightarrow$ $\forall{}p\in{}F\inv(q)$, $F$ is a submersion at $p$.
	\end{itemize}\up\n
}

\par\noindent
Observe: If $N=\R$, then either $F_{*,p}:T_{p}M\to{}T_{F(p)}\R\cong\R$ is onto, or it is zero. So our definition is consistent with that used in single-variable calculus.\n

\par\noindent
Observe: If $q\in{}N$ satisfies $F\inv(q)=\emptyset$, then $q$ is a regular value.\n

\thm{
	(Normal Form Theorem) Assume $F:M\to{}N$ is a submersion at $p\in{}M$. Let $m=\dim{}M$ and $n=\dim{}N$. Then there are coordinate charts $(U,\phi)$ around $p$ and $(V,\psi)$ around $F(p)$, with $U\subseteq{}F\inv(V)$, such that for\break $\tilde{F}=\psi\of{}F\of\phi\inv$, $\tilde{F}(r^{1},\ldots,r^{m})=(r^{1},\ldots,r^{n})$.
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			M \arrow[r,"F"] \arrow[d,"\phi"] \& N \arrow[d,"\psi"]\\
			\R^{m}\supseteq\phi(U) \arrow[r,"\tilde{F}"] \& \psi(V)\subseteq\R^{n}
		\end{tikzcd}
	\]
	Proof: Start with any coordinates $\phi=(x^{1},\ldots,x^{m})$, $\psi=(y^{1},\ldots,y^{n})$. Wrie $F=(F^{1},\ldots,F^{n})$, with each $F^{i}=y^{i}\of{}F$. Then the Jacobian of $F$ at $p$ is
	\[
		\begin{bmatrix}
			\pd{F^{1}}{x^{1}}(p) & \cdots & \pd{F^{1}}{x^{m}}(p)\\
			\vdots & & \vdots\\
			\pd{F^{n}}{x^{1}}(p) & \cdots & \pd{F^{n}}{x^{m}}(p)
		\end{bmatrix}
	\]
	We know $J$ has full rank. So by permuting the $x^{j}$'s, WOLOG $J:(M\;*)$ where $M$ is $n\times{}n$ and full rank.\n
	Define $\tilde\phi=(F^{1},\ldots,F^{n},x^{n+1},\ldots,x^{m})$. We claim that $\tilde{\phi}:U\to\R^{m}$ is a local diffeomorphism at $p$, and these are the desired coordinates.\n
	Well, the Jacobian of $\tilde\phi$ in the $x$ coordinates is
	\[
		\begin{bmatrix}
			M & *\\
			0 & I_{m-n}
		\end{bmatrix}
	\]
	because $\pd{x^{i}}{x^{j}}=\delta_{ij}$. And this matrix is invertible because $M$ is.\proven
}

\defn{
	Let $M$ be a manifold. Let $S\subseteq{}M$. Then $S$ is a (\u{regular}) \u{submanifold} of $M$ iff there is some $k\in\N$ \st{} $\forall{}p\in{}S$, there is a chart $(U,\phi)$ of $M$ with $p\in{}U$, such that $S\cap{}U=\set{q\in{}U\mid{}x^{j}(q)=0\ptxt{ for }j\in\set{k+1,\ldots,m}}$. Such a coordinate system $(U,\phi)$ is \u{adapted} to the submanifold and defines an \u{induced chart} on $S$ by $\phi_{S}:S\cap{}U\to\R^{k}$, where $\phi_{S}=\restr{(x^{1},\ldots,x^{k})}{S\cap{}U}$.\n
}

\par\noindent
Claim: $\phi_{S}:S\cap{}U\to\phi(S\cap{}U)\osubseteq\R^{k}$ is a homeomorphism with respect to the subspace topology.\n

\par\noindent
Claim: With the subspace topology and an atlas of adapted charts, $S$ inherits a $C^{\infty}$ structure. That is, any two charts on $S$ arising from adapted coordinate systems are $C^{\infty}$-compatible.\n

\par\noindent
The idea of the proof is given two charts $(U,\phi)$ and $(V,\psi)$, $\psi_{S}\of(\phi_{S})\inv$ is the restriction of $\psi\of\phi\inv$, so the former is $C^{\infty}$.\n

\par\noindent
We also need to check the point-set topology conditions: Hausdorff and second-countable. But they follow directly from the subspace topology.\n

\par\noindent
Can a chart on the submanifold be extended to a chart on the original manifold? Yes!\n
Observe: If we start with $(x^{1},\ldots,x^{k}):W\to\R^{k}$, a chart of $S$, then there is a chart $\phi$ on $U\osubseteq{}M$ \st{} $W=U\cap{}S$ and $(x^{1},\ldots,x^{k})=\phi_{S}$.\n

\prop{
	If $F:\R^{N}\to\R^{N-k}$ and $0$ is a regular value of $F$, then $F\inv(0)$ is a regular submanifold of $\R^{N}$.\nn
	Proof: WOLOG $S$ is the graph of $G:A\to\R^{N-k}$, with $A\osubseteq\R^{k}$. $S=\set{(a,G(a))\mid{}a\in{}A}$.\n
	Define $x^{i}=x^{i}$ for $1\le{}i\le{}k$, and $x^{j}=x^{j}-G^{j}(x^{1},\ldots,x^{k})$ for $k+1\le{}j\le{}N$, with $G=(G^{1},\ldots,G^{N-k})$. These are adapted coordinates!\proven
}

\par\noindent
The same proof, using the normal form theorem, gives:
\thm{
	If $q\in{}N$ is a regular value of $F:M\to{}N$, then $S=F\inv(q)\subseteq{}M$ is a regular submanifold.\n
}
