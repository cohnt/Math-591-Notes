\ex{
	(of germs)\n
	Let $p\in\R^{n}$. Then $C^{\infty}_{p}(\R^{n})=\set{(U,f)\mid{}p\in{}U\overset{\ptxt{open}}{\subseteq}\R^{n},f:U\to\R\;C^{\infty}}/\sim$.\n
}

\par\noindent
Observe: There is a well-defined map
\[
	\map{C^{\infty}_{p}(\R^{n})}{\R[[r]],r=(r^{1},\ldots,r^{n})}{{}[f]}{f(p)+\sum_{j=1}^{n}(r^{j}-r_{0}^{j})\pd{f}{r^{j}}(p)+\cdots}
\]
where $\R[[r]]$ is the set of formal power series in the $r^{i}$ variables, and $[f]$ maps to the Taylor series of $f$ at $p=(r_{0}^{1},\ldots,r_{0}^{n})$.\n
Why is this well defined? Well, if $[f]=[g]$, then $f$ and $g$ agree on a neighborhood of $p$.\n

\prop{
	\begin{enumerate}[topsep=0pt, itemsep=0pt, label=(\arabic*), leftmargin=4\parindent]
		\item This map is a surjection, i.e., any formal power series is the Taylor series of some smooth functions.\n
		\item This map is \i{not} injective, i.e., there exist $C^{\infty}$ functions $f$ defined near $p$ \st{} $\forall\alpha$ multi-indices, $\pd{^{\alpha}f}{r^{\alpha}}(p)=0$, but $f$ is not zero near $p$.
	\end{enumerate}\up\n
	This is just an FYI -- we're not going to use this for a while.\n
}

\par\noindent
Now, back to manifolds...\n
Let $M$ be a $C^{\infty}$ manifold, and $p\in{}M$. We defined $T_{p}M=\set{\ptxt{all derivations }D:C^{\infty}_{p}(M)\to\R}$.\n

\ex{
	(of derivations)\n
	Start with a curve $\gamma:(t_{0}-\varepsilon,t_{0}+\varepsilon)\to{}M$ smooth \st{} $\gamma(t_{0})=p$. Define
	\[
		\map{\dot\gamma(t_{0}):C^{\infty}_{p}(M)}{\R}{{}[f]}{\restr{\frac{d}{dt}(f\of\gamma)(t)}{t=t_{0}}}
	\]
	It's easy to check that $\dot\gamma(t_{0})$ is a derivation (by calc III stuff). Note that this defines $\dot\gamma(t_{0})\in{}T_{\gamma(t_{0})}M$.\n
	We will see today that \i{all} derivations are of this form.\n
}

\par\noindent
Observe: In the case where $M\subseteq\R^{N}$ is a local graph, then $\gamma:(t_{0}-\varepsilon,t_{0}+\varepsilon)\to{}M\hookrightarrow\R^{N}$ can be interepreted as a smooth curve in $\R^{N}$. $\dot\gamma(t_{0})$ was defined in calc III as an element in $\R^{N}$. These definitions are consistent! But our definition doesn't need an ambient space.

\subsection*{Introducting Local Coordinates and Partial Derivatives}

\par\noindent
Let $p\in{}U\subseteq{}M$, with $\phi:U\to\R^{N}$ a chart. Then we use the notation $r^{i}:\R^{N}\to\R$ is the $i$th component/coordinate.\n
We say $x^{i}=r^{i}\of\phi:U\to\R$, so we can write $\phi=(x^{1},x^{2},\ldots,x^{N})$. (The $x^{i}$s are defined on $U\subseteq{}M$.)\n

\defn{
	Given $f:U\to\R$ smooth, $p\in{}U$,
	\[
		\pd{f}{x^{i}}(p)=\pd{}{r^{i}}(f\of\phi\inv)[\phi(p)]\in\R
	\]
}

\par\noindent
Some notation: we write $f_{\phi}\eqdef{}f\of\phi\inv$. Observe that $\pd{f}{x^{i}}=\pd{f_{\phi}}{x^{i}}\of\phi$.\n

\newpage
\lemma{
	$\forall{}i\in\set{1,\ldots,n}$, the map
	\[
		\restr{\pd{}{x^{i}}}{p}:C^{\infty}_{p}(M)\ni[f]\mapsto\pd{f}{x^{i}}(p)
	\]
	is a derivation at $p$, and moreover,
	\[
		\Phi\eqdef\set{\restr{\pd{}{x^{1}}}{p},\ldots,\restr{\pd{}{x^{n}}}{p}}
	\]
	is a basis (over $\R$) of $T_{p}M$.\n
}

\par\noindent
Observe: $\restr{\pd{}{x^{i}}}{p}$ are velocities of curves. Let $\phi(p)=(r_{0}^{1},\ldots,r_{0}^{n})$. Then if $\gamma_{i}:t\mapsto\phi\inv(r_{0}^{1},\ldots,r_{0}^{i}+t,\ldots,r_{0}^{n})$ for $t\in(-\varepsilon,\varepsilon)$, we claim that $\dot\gamma(p)=\restr{\pd{}{x^{i}}}{p}$.\n

\par\noindent
To actually prove that $\Phi$ is a basis, we need:\n

\thm{
	Let $g$ be a $C^{\infty}$ function defined in a neighborhood of a point $r_{0}\in\R^{n}$. Then $\exists{}g_{ij}$, with $i,j\in\set{1,\ldots,n}$, that is smooth and defined near $r_{0}$, such that $\forall{}r\in\dom(g)$,
	\[
		g(r)=\underbrace{g(0)+\sum_{j=1}^{n}(r^{j}-r_{0}^{j})\pd{g}{r^{j}}}_{\ptxt{First degree Taylor polynomial}}+\underbrace{\frac{1}{2}\sum_{i,j=1}^{n}(r^{i}-r_{0}^{i})(r^{j}-r_{0}^{j})\cdot{}g_{ij}(r)}_{\substack{\ptxt{``An interesting way of writing}\\ \ptxt{the remainder''}}}
	\]
	(Moreover, $g_{ij}(r_{0})=\pd{^{2}g}{r^{i}\partial{}r^{j}}(r_{0})$.)\n
}

\par\noindent
Let $D\in{}T_{p}M$, $[f]\in{}C^{\infty}_{p}(M)$. Apply this to $g=f_{\phi}$. We claim that
\[
	D[f]=\sum_{i=1}^{n}\pd{f}{x^{i}}(p)\cdot{}D([x^{i}])
\]
This implies
\[
	D=\sum_{i=1}^{n}D([x^{i}])\restr{\pd{}{x^{i}}}{p}
\]
