\par\noindent
Observe:
\begin{enumerate}[label=(\arabic*)]
	\item $\dim{}O(n)=\dim(\ptxt{ambient})-\dim(\Symm(n,\R))=n^{2}-\frac{n(n+1)}{2}=\frac{n(n-1)}{2}$.
	\item $\ker{}F'(I)=\set{M\in\Mat(n,\R)\mid{}M+M\tpose=0}=\set{\ptxt{skew-symmetric matrices}}$. This is the tangent space to $O(n)$ at $I$.\n
	\n
	Similarly, $U(n)=\set{g\in\Mat(n,\C)\mid{}g\inv=\conj{g}\tpose}$ has a $C^{\infty}$ structure, as well as $SU(n)=\set{g\in{}U\mid{}\det{}g=1}$.\n
	($g\in{}U(n)\Rightarrow\det(g)\in{}S^{1}$, i.e., $\abs{\det{}g}=1$.)\n
	\item $O(n)$, in fact, has $2$ connected components, as $g\in{}O(n)\Rightarrow\underbrace{\abs{\det{}g}}_{\in\R}=1\Rightarrow\det{}g=\pm{}1$.
\end{enumerate}

\defn{
	$SO(n)=\set{g\in{}O(n)\mid{}\det{}g=1}$ is a subgroup of $O(n)$.\n
	$O(n)=SO(n)\cup\set{g\in{}O(n)\mid{}\det{}g=-1}$.\n
}

\par\noindent
More examples: $SL(n,\R)=\set{g\in\GL(n,\R)\mid{}\det{}g=1}$.\n
Some facts: $U(n)$ and $O(n)$ are compact, whereas $SL(n,\R)$ is not.\n

\par\noindent
More examples can be constructed by:
\begin{itemize}[label=-]
	\item Cartesian products: If $M$ and $N$ are $C^{\infty}$ manifolds, then $M\times{}N$ has a natural smooth structure. Charts on $M\times{}N$ are just $(U\times{}V,\phi\times\psi)$, where $(U,\phi)$ is a chart on $M$ and $(V,\psi)$ is a chart on $N$.\n
	For example, the $n$th torus $\underbrace{S^{1}\times\cdots\times{}S^{1}}_{n}$.
	\item Covering maps of $C^{\infty}$ manifolds: Let $M$ be a $C^{\infty}$ manifold. A covering map on $M$ is $f:\tilde{M}\to{}M$ (with $\tilde{M}$ a topological space) such that $\forall{}p\in{}M$, $\exists{}U\ni{}p$ open \st{} $F\inv(U)=\bigcup_{i\in{}I\ptxt{ finite}}U_{i}$, with $U_{i}\subseteq\tilde{M}$ open \st{} $\forall{}i$, $\restr{F}{U_{i}}:U_{i}\overset{\cong}{\to}V$ is a homeomorphism.
\end{itemize}
Then:\n

\thm{
	$\tilde{M}$ has a unique $C^{\infty}$ manifold structure \st{} $F$ is locally a diffeomorphism (isomorphism).\n
}

\thm{
	$SO(n)$ has a double cover (a $2$-to-$1$ covering space), $\operatorname{Spin}(n)\overset{\ptxt{2-1}}{\to}SO(n)$. $\operatorname{Spin}(n)$ has a group structure.\n
}

\ex{
	(of a covering map)
	\[
		\map{\R}{S^{1}}{x}{e^{ix}}
	\]
}

\defn{
	Let $M$ be a $C^{\infty}$ manifold, and $f:M\to\R$, $p\in{}M$. Then \underline{$f$ is $C^{\infty}$ at $p$} if there's a chart $(U,\phi)$ of $M$ such that $p\in{}U$ and $f\of\phi\inv:\phi(U)\to\R$ is $C^{\infty}$.\n
}

\par\noindent
Observe: A chart $(U,\phi)$ one a $C^{\infty}$ manifold is also called a coordinate system. We'll often write $\phi=(x^{1},\ldots,x^{n})$, where $x^{i}:U\to\R$ is the $i$th component of $\phi$, i.e., a coordinate function.\n

\par\noindent
Observe: In the definition above, $f$ only needs to be defined in a neighborhood of $p$.\n

\defn{
	Let $M$ be a $C^{\infty}$ manifold, $f:M\to\R$ is \underline{smooth} iff $\forall{}p\in{}M$, $f$ is smooth at $p$.\n
}

\newpage
\lemma{
	$f:M\to\R$ is smooth iff $\forall(U,\phi)$ smooth chart of $M$, $f\of\phi\inv:\phi(U)\to\R$ is smooth.\n
	Proof: (see \sectionSymbol{}6 for full details)\n
	$\Leftarrow$ is immediate\n
	$\Rightarrow$ is based on the fact that $f$ smooth $\Rightarrow$ $\forall{}p\in{}M$, there's a chart $(V,\psi)$ around $p$ \st{} $f\of\psi\inv$ is smooth.\n
}

\ex{
	Let $M\subseteq\R^{N}$ be a local graph. $M=F\inv(0)$, $0$ is a regular value of $F$.\n
	If $\tilde{f}:\R^{N}\to\R$ is smooth, then $f=\restr{\tilde{f}{M}}:M\to\R$ is smooth.\n
	Proof: There are charts on $M$ $(U,\phi)$ \st{} $\phi\inv(x')=(x',G(x'))$ after permuting coordinates ($G$ is a graph function). Then $(f\of\phi\inv)(x')=\tilde{f}(x',G(x'))$, and this is $C^{\infty}$.\proven
}
